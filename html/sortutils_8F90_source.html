1035 ! Insert partitioning element
1036 arr1d(l + 1) = arr1d(j)
1037 arr1d(j) = a

1053 END IF

1055 IF (r - i + 1 >= j - l) THEN
1056 stack(ist) = r
1057 stack(ist - 1) = i
1058 r = j - 1
1059 ELSE
1060 stack(ist) = j - 1
1061 stack(ist - 1) = l
