68 SUBROUTINE openfileforread(lun, fileName, errorIO)

70 USE pahm_global

72 IMPLICIT NONE
73
74 ! Global variables
75 INTEGER, INTENT(IN) :: lun ! fortran logical unit number
76 CHARACTER(LEN=*), INTENT(IN) :: fileName ! full pathname of file
77 INTEGER, INTENT(OUT) :: errorIO ! zero if the file opened successfully
78
79 ! Local variables
80 LOGICAL :: fileFound ! .TRUE. if the file is present
81 CHARACTER(LEN=LEN(fileName)) :: tmpFileName ! full pathname of file
82
83 CALL setmessagesource("OpenFileForRead")
84
85 errorio = 0
86
87 tmpfilename = adjustl(filename)
88
89 ! Check to see if file exists
90 WRITE(scratchmessage, '("Searching for file to open on unit ", i0, "...")') lun
91 CALL logmessage(info, trim(scratchmessage))
92
93 INQUIRE(file=trim(filename), exist=filefound)
94 IF (.NOT. filefound) THEN
95 WRITE(scratchmessage, '("The file : ", a, " was not found.")') trim(tmpfilename)
96 CALL allmessage(info, scratchmessage)
97
98 errorio = 1
99
100 CALL unsetmessagesource()
101
102 RETURN ! file not found
103 ELSE
104 WRITE(scratchmessage, '("The file : ", a, " was found. The file will be opened.")') trim(tmpfilename)
105
106 CALL logmessage(info, trim(scratchmessage))
107 END IF
108
109 ! Open existing file
110 OPEN(unit=lun, file=trim(tmpfilename), status='OLD', action='READ', iostat=errorio)
111 IF (errorio /= 0) THEN
112 WRITE(scratchmessage, '("Could not open the file: ", a, ".")') trim(tmpfilename)
113
114 CALL allmessage(error, trim(scratchmessage))
115
116 CALL unsetmessagesource()
117
118 RETURN ! file found but could not be opened
119 ELSE
120 WRITE(scratchmessage, '("The file ", a, " was opened successfully.")') trim(tmpfilename)
121
122 CALL logmessage(info, trim(scratchmessage))
123 END IF
124
125 CALL unsetmessagesource()
126
127 RETURN

265 npnts = loadintvar(nval, realval, 1, ivalout)

268 ELSE
269 writeparams = .false.

273 CASE ('NBTRFILES')
274 npnts = loadintvar(nval, realval, 1, ivalout)
275 nbtrfiles = ivalout(1)
276 IF (nbtrfiles > 0) THEN
277 ALLOCATE(besttrackfilename(nbtrfiles))
278 besttrackfilename = blank
279 END IF
280 gotnbtrfiles = .true.
281
282 !----- CASE
283 CASE ('BESTTRACKFILENAME')
284 IF (.NOT. gotnbtrfiles) THEN
285 WRITE(scratchmessage, fmtparnotfound) 'nBTrFiles', '(add the "nBTrFiles" keyword before "bestTrackFileName").'
286 CALL allmessage(error, scratchmessage)
287 ELSE
288 IF (ALLOCATED(besttrackfilename)) THEN
289 tmpstr = adjustl(charval(nval))
290 IF (trim(tmpstr) == '') THEN
291 nval = nval - 1
292 ELSE
293 IF (nval <= nbtrfiles) THEN ! because bestTrackFileName has been allocated this way above
294 numbtfiles = numbtfiles + 1
295 besttrackfilename(numbtfiles) = trim(tmpstr)
296 besttrackfilenamespecified = .true.
297 END IF
298 END IF
299 END IF
300 END IF
301
302 !----- CASE
303 CASE ('MESHFILETYPE')
304 IF (nval == 1) THEN
305 meshfiletype = trim(adjustl(charval(nval)))
306 ELSE
307 IF (trim(adjustl(meshfiletype)) == '') THEN
308 meshfiletype = trim(adjustl(charval(nval)))
309 END IF
310 END IF
311
312 !----- CASE
313 CASE ('MESHFILENAME')
314 IF (nval == 1) THEN
315 meshfilename = trim(adjustl(charval(nval)))
316 ELSE
317 IF (trim(adjustl(meshfilename)) == '') THEN
318 meshfilename = trim(adjustl(charval(nval)))
319 END IF
320 END IF
321 IF (trim(adjustl(meshfilename)) /= '') meshfilenamespecified = .true.
322
323 !----- CASE
324 CASE ('MESHFILEFORM')
325 IF (nval == 1) THEN
326 meshfileform = trim(adjustl(charval(nval)))
327 ELSE
328 IF (trim(adjustl(meshfileform)) == '') THEN
329 meshfileform = trim(adjustl(charval(nval)))
330 END IF
331 END IF
332
333 !----- CASE
334 CASE ('GRAVITY')
335 npnts = loadrealvar(nval, realval, 1, rvalout)
336 gravity = rvalout(1)
337
338 !----- CASE
339 CASE ('RHOWATER')
340 npnts = loadrealvar(nval, realval, 1, rvalout)
341 rhowater = rvalout(1)
342
343 !----- CASE
344 CASE ('RHOAIR')
345 npnts = loadrealvar(nval, realval, 1, rvalout)
346 rhoair = rvalout(1)
347
348 !----- CASE
349 CASE ('BACKGROUNDATMPRESS')
350 npnts = loadrealvar(nval, realval, 1, rvalout)
351 backgroundatmpress = rvalout(1)
352
353 !----- CASE
354 CASE ('BLADJUSTFAC')
355 npnts = loadrealvar(nval, realval, 1, rvalout)
356 bladjustfac = rvalout(1)
357
358 !----- CASE
359 CASE ('REFDATETIME')
360 IF (nval == 1) THEN
361 refdatetime = trim(adjustl(charval(nval)))
362 ELSE
363 IF (trim(adjustl(refdatetime)) == '') THEN
364 refdatetime = trim(adjustl(charval(nval)))
365 END IF
366 END IF
367
368 CALL splitdatetimestring(refdatetime, refyear, refmonth, refday, refhour, refmin, refsec)
369
370 IF ((refyear == imissv) .OR. (refmonth <= 0) .OR. (refday <= 0)) THEN
371 refdate = imissv
372 ELSE
373 refdate = joindate(refyear, refmonth, refday)
374 END IF
375 reftime = joindate(refhour, refmin, refsec)
376 refdatespecified = .true.
377
378 !----- CASE
379 CASE ('UNITTIME')
380 IF (begdatespecified .OR. begsimspecified .OR. &
381 enddatespecified .OR. endsimspecified) THEN
382 scratchmessage = 'add the "unitTime" keyword before the ' // &
383 '"begDateTime"/"begTime" and "endDateTime"/"endTime" keywords'
384 CALL allmessage(error, scratchmessage)
385 CALL terminate()
386 ELSE
387 IF (nval == 1) THEN
388 tmpcharval = touppercase(adjustl(charval(nval)))
389 unittime = tmpcharval(1:1)
390 ELSE
391 IF (trim(adjustl(unittime)) == '') THEN
392 tmpcharval = touppercase(adjustl(charval(nval)))
393 unittime = tmpcharval(1:1)
394 END IF
395 END IF
396 END IF
397
398 !----- CASE
399 CASE ('BEGDATETIME')
400 IF (begdatespecified .OR. begsimspecified) THEN
401 scratchmessage = 'Only one of "begDateTime" or "begSimTime" can be specified'
402 CALL allmessage(error, scratchmessage)
403
404 begdatespecified = .false.
405 ELSE
406 IF (.NOT. refdatespecified) THEN
407 scratchmessage = 'Add the "refDateTime" keyword before "begDateTime").'
408 CALL allmessage(error, scratchmessage)
409 ELSE
410 IF (nval == 1) THEN
411 begdatetime = trim(adjustl(charval(nval)))
412 ELSE
413 IF (trim(adjustl(begdatetime)) == '') THEN
414 begdatetime = trim(adjustl(charval(nval)))
415 END IF
416 END IF
417
418 CALL splitdatetimestring(begdatetime, begyear, begmonth, begday, beghour, begmin, begsec)
419
420 IF ((begyear == imissv) .OR. (begmonth <= 0) .OR. (begday <= 0)) THEN
421 begdate = imissv
422 ELSE
423 begdate = joindate(begyear, begmonth, begday)
424 END IF
425 begtime = joindate(beghour, begmin, begsec)
426
427 CALL timeconv(begyear, begmonth, begday, beghour, begmin, begsec, mdbegsimtime)
428 begsimtime = mdbegsimtime * gettimeconvsec(unittime, 1)
429
430 begdatetime = datetime2string(begyear, begmonth, begday, beghour, begmin, begsec)
431
432 begdatespecified = .true.
433 begsimspecified = .true.
434 END IF
435 END IF
436
437 !----- CASE
438 CASE ('ENDDATETIME')
439 IF (enddatespecified .OR. endsimspecified) THEN
440 scratchmessage = 'Only one of "endDateTime" or "endSimTime" can be specified'
441 CALL allmessage(error, scratchmessage)
442
443 enddatespecified = .false.
444 ELSE
445 IF (.NOT. refdatespecified) THEN
446 scratchmessage = 'Add the "refDateTime" keyword before "endDateTime").'
447 CALL allmessage(error, scratchmessage)
448 ELSE
449 IF (nval == 1) THEN
450 enddatetime = trim(adjustl(charval(nval)))
451 ELSE
452 IF (trim(adjustl(enddatetime)) == '') THEN
453 enddatetime = trim(adjustl(charval(nval)))
454 END IF
455 END IF
456
457 CALL splitdatetimestring(enddatetime, endyear, endmonth, endday, endhour, endmin, endsec)
458
459 IF ((endyear == imissv) .OR. (endmonth <= 0) .OR. (endday <= 0)) THEN
460 enddate = imissv
461 ELSE
462 enddate = joindate(endyear, endmonth, endday)
463 END IF
464 endtime = joindate(endhour, endmin, endsec)
465
466 CALL timeconv(endyear, endmonth, endday, endhour, endmin, endsec, mdendsimtime)
467 endsimtime = mdendsimtime * gettimeconvsec(unittime, 1)
468
469 enddatetime = datetime2string(endyear, endmonth, endday, endhour, endmin, endsec)
470
471 enddatespecified = .true.
472 endsimspecified = .true.
473 END IF
474 END IF
475
476 !----- CASE
477 CASE ('OUTDT')
478 npnts = loadrealvar(nval, realval, 1, rvalout)
479 outdt = rvalout(1)
480 mdoutdt = fixnearwholereal(outdt * gettimeconvsec(unittime), closetol)
481
482 !----- CASE
483 CASE ('BEGSIMTIME')
484 IF (begdatespecified .OR. begsimspecified) THEN
485 scratchmessage = 'Only one of "begDateTime" or "begSimTime" can be specified'
486 CALL allmessage(error, scratchmessage)
487
488 begsimspecified = .false.
489 ELSE
490 IF (.NOT. refdatespecified) THEN
491 scratchmessage = 'Add the "refDateTime" keyword before "begSimTime").'
492 CALL allmessage(error, scratchmessage)
493 ELSE
494 npnts = loadrealvar(nval, realval, 1, rvalout)
495 begsimtime = rvalout(1)
496
497 mdbegsimtime = begsimtime * gettimeconvsec(unittime)
498
499 jday = (mdbegsimtime * gettimeconvsec('D', 1)) + gregtojulday(refyear, refmonth, refday, refhour, refmin, refsec)
500 CALL juldaytogreg(jday, begyear, begmonth, begday, beghour, begmin, begsec)
501 begdatetime = datetime2string(begyear, begmonth, begday, beghour, begmin, begsec)
502
503 begdatespecified = .true.
504 begsimspecified = .true.
505 END IF
506 END IF
507
508 !----- CASE
509 CASE ('ENDSIMTIME')
510 IF (enddatespecified .OR. endsimspecified) THEN
511 scratchmessage = 'Only one of "endDateTime" and "endSimTime" can be specified'
512 CALL allmessage(error, scratchmessage)
513
514 endsimspecified = .false.
515 ELSE
516 IF (.NOT. refdatespecified) THEN
517 scratchmessage = 'Add the "refDateTime" keyword before "endSimTime").'
518 CALL allmessage(error, scratchmessage)
519 ELSE
520 npnts = loadrealvar(nval, realval, 1, rvalout)
521 endsimtime = rvalout(1)
522
523 mdendsimtime = endsimtime * gettimeconvsec(unittime)
524
525 jday = (mdendsimtime * gettimeconvsec('D', 1)) + gregtojulday(refyear, refmonth, refday, refhour, refmin, refsec)
526 CALL juldaytogreg(jday, endyear, endmonth, endday, endhour, endmin, endsec)
527 begdatetime = datetime2string(endyear, endmonth, endday, endhour, endmin, endsec)
528
529 enddatespecified = .true.
530 endsimspecified = .true.
531 END IF
532 END IF
533
534 !----- CASE
535 CASE ('OUTFILENAME')
536 IF (nval == 1) THEN
537 outfilename = trim(adjustl(charval(nval)))
538 ELSE
539 IF (trim(adjustl(outfilename)) == '') THEN
540 outfilename = trim(adjustl(charval(nval)))
541 END IF
542 END IF
543 IF (trim(adjustl(outfilename)) /= '') outfilenamespecified = .true.
544
545 !----- CASE
546 CASE ('NCVARNAM_PRES')
547 IF (nval == 1) THEN
548 ncvarnam_pres = trim(adjustl(charval(nval)))
549 ELSE
550 IF (trim(adjustl(ncvarnam_pres)) == '') THEN
551 ncvarnam_pres = trim(adjustl(charval(nval)))
552 END IF
553 END IF
554
555 !----- CASE
556 CASE ('NCVARNAM_WNDX')
557 IF (nval == 1) THEN
558 ncvarnam_wndx = trim(adjustl(charval(nval)))
559 ELSE
560 IF (trim(adjustl(ncvarnam_wndx)) == '') THEN
561 ncvarnam_wndx = trim(adjustl(charval(nval)))
562 END IF
563 END IF
564
565 !----- CASE
566 CASE ('NCVARNAM_WNDY')
567 IF (nval == 1) THEN
568 ncvarnam_wndy = trim(adjustl(charval(nval)))
569 ELSE
570 IF (trim(adjustl(ncvarnam_wndy)) == '') THEN
571 ncvarnam_wndy = trim(adjustl(charval(nval)))
572 END IF
573 END IF
574
575 !----- CASE
576 CASE ('NCSHUFFLE')
577 npnts = loadintvar(nval, realval, 1, ivalout)
578 ncshuffle = ivalout(1)
579
580 !----- CASE
581 CASE ('NCDEFLATE')
582 npnts = loadintvar(nval, realval, 1, ivalout)
583 ncdeflate = ivalout(1)
584
585 !----- CASE
586 CASE ('NCDLEVEL')
587 npnts = loadintvar(nval, realval, 1, ivalout)
588 ncdlevel = ivalout(1)
589
590 !----- CASE
591 CASE ('MODELTYPE')
592 npnts = loadintvar(nval, realval, 1, ivalout)
593 modeltype = ivalout(1)
594
595 !----- CASE
596 CASE DEFAULT
597 ! Do nothing
598 END SELECT
599 END IF
600 END DO
601
602 10 WRITE(lun_screen, '("Error while processing line: ", a, " in file: ", a)') &
603 trim(adjustl(inpline)), trim(tmpfilename)
604
605 CLOSE(iunit)
606 stop
607
608 20 CLOSE(iunit)
609
610 WRITE(lun_screen, '(a)') 'Finished processing the input fields from the control file ...'

613 !--- CHECK INPUT VARIABLES AND SET DEFAULTS

623 END IF

626
627 CALL unsetmessagesource()

629 END SUBROUTINE readcontrolfile

869 INTEGER FUNCTION parseline(inpLine, outLine, keyWord, nVal, cVal, rVal) RESULT(myStatus)

899 ! - 45

903 ! \ 92

915 ! values. Find equal sign [CHAR(61)].

919 status = 1

922 lstr = index(line, char(61), back = .true.) + 1
923
924 ! Determine if KEYWORD is followed by double equal sign (==) indicating
925 ! nested parameter.
926 IF ((lstr - kend) == 3) nested = .true.
927
928 ! Extract KEYWORD, trim leading and trailing blanks.
929 kextract = .false.
930 IF (kend > 0) THEN
931 lend = lenline
932 keyword = line(kstr:kend)
933 nval = 0
934 kextract = .true.
935 ELSE
936 lstr = 1
937 lend = lenline
938 kextract = .true.
939 END IF
940
941 ! Extract parameter values string. Remove continuation symbol
942 ! [CHAR(92)=\] or multi-line value [CHAR(124)=|], if any. Trim
943 ! leading trailing blanks.
944 IF (kextract) THEN
945 icont = index(line, char(92 ), back = .false.)
946 ipipe = index(line, char(124), back = .false.)
947 IF (icont > 0) lend = icont - 1
948 IF (ipipe > 0) lend = ipipe - 1
949 vstring = adjustl(line(lstr:lend))
950 lval = len_trim(vstring)

957 nval = nval + 1
958 cval(nval) = vstring(1:lval)

961 CASE ('LOGFILENAME')
962 nval = nval + 1
963 cval(nval) = vstring(1:lval)
964 isstring = .true.
965
966 CASE ('BESTTRACKFILENAME')
967 nval = nval + 1
968 cval(nval) = vstring(1:lval)
969 isstring = .true.
970
971 CASE ('MESHFILENAME')
972 nval = nval + 1
973 cval(nval) = vstring(1:lval)
974 isstring = .true.
975
976 CASE ('MESHFILETYPE')
977 nval = nval + 1
978 cval(nval) = vstring(1:lval)
979 isstring = .true.
980
981 CASE ('MESHFILEFORM')
982 nval = nval + 1
983 cval(nval) = vstring(1:lval)
984 isstring = .true.
985
986 CASE ('REFDATETIME')
987 nval = nval + 1
988 cval(nval) = vstring(1:lval)
989 isstring = .true.
990
991 CASE ('UNITTIME')
992 nval = nval + 1
993 cval(nval) = vstring(1:lval)
994 isstring = .true.
995
996 CASE ('BEGDATETIME')
997 nval = nval + 1
998 cval(nval) = vstring(1:lval)
999 isstring = .true.
1000
1001 CASE ('ENDDATETIME')
1002 nval = nval + 1
1003 cval(nval) = vstring(1:lval)
1004 isstring = .true.
1005
1006 CASE ('OUTFILENAME')
1007 nval = nval + 1
1008 cval(nval) = vstring(1:lval)
1009 isstring = .true.
1010
1011 CASE ('NCVARNAM_PRES')
1012 nval = nval + 1
1013 cval(nval) = vstring(1:lval)
1014 isstring = .true.
1015
1016 CASE ('NCVARNAM_WNDX')
1017 nval = nval + 1
1018 cval(nval) = vstring(1:lval)
1019 isstring = .true.
1020
1021 CASE ('NCVARNAM_WNDY')
1022 nval = nval + 1
1023 cval(nval) = vstring(1:lval)
1024 isstring = .true.
1025
1026 CASE DEFAULT
1027 ! For every other KEYWORD except the above.
1028 ! Check if there is a multiplication symbol [CHAR(42)=*] in the variable
1029 ! string indicating repetition of input values.
1030 nmul = 0
1031 DO i = 1, lval
1032 IF (vstring(i:i) == char(42)) THEN
1033 nmul = nmul + 1
1034 imul(nmul) = i
1035 END IF
1036 END DO
1037 ic = 1
1038
1039 ! Check for blank spaces [CHAR(32)=' '] between entries and decode.
1040 is = 1
1041 ie = lval
1042 iblank = 0
1043 decflag = .false.
1044 DO i = 1, lval
1045 IF (vstring(i:i) == char(32)) THEN
1046 IF (vstring(i + 1:i + 1) /= char(32)) decflag = .true.
1047 iblank = i
1048 ELSE
1049 ie = i
1050 END IF
1051 IF (decflag .OR. (i == lval)) THEN
1052 nval = nval + 1
1053
1054 ! Processing numeric values. Check starting character to determine
1055 ! if numeric or character values. It is possible to have both when
1056 ! processing repetitions via the multiplication symbol.
1057 schar = ichar(vstring(is:is))
1058 IF (((48 <= schar) .AND. (schar <= 57)) .OR. (schar == 43) .OR. (schar == 45)) THEN
1059 IF ((nmul > 0) .AND. (is < imul(ic)) .AND. (imul(ic) < ie)) THEN
1060 READ(vstring(is:imul(ic) - 1), *) copies
1061 schar = ichar(vstring(imul(ic) + 1:imul(ic) + 1))
1062 IF ((43 <= schar) .AND. (schar <= 57)) THEN
1063 READ(vstring(imul(ic) + 1:ie), *) rval(nval)
1064 DO j = 1, copies - 1
1065 rval(nval + j) = rval(nval)
1066 END DO
1067 ELSE
1068 string = vstring(imul(ic) + 1:ie)
1069 lens = len_trim(string)
1070 cval(nval) = string(1:lens)
1071 DO j = 1, copies - 1
1072 cval(nval + j) = cval(nval)
1073 END DO
1074 END IF
1075 nval = nval + copies - 1
1076 ic = ic + 1
1077 ELSE
1078 string = vstring(is:ie)
1079 lens = len_trim(string)
1080 !READ(string(1:lEns), *) rVal(nVal)
1081 READ(string(1:lens), *, iostat=ierr) rval(nval)
1082 IF (ierr /= 0) THEN
1083 WRITE(*, *) '#### ERROR :: Cannot interpret string ', string(1:lens), &
1084 ' as a REAL number.'
1085 END IF
1086 END IF
1087 ELSE
1088
1089 ! Processing character values (logicals and strings).
1090 IF ((nmul > 0) .AND. (is < imul(ic)) .AND. (imul(ic) < ie)) THEN
1091 READ(vstring(is:imul(ic) - 1), *) copies
1092 cval(nval) = vstring(imul(ic) + 1:ie)
1093 DO j = 1, copies - 1
1094 cval(nval + j) = cval(nval)
1095 END DO
1096 nval = nval + copies - 1
1097 ic = ic + 1
1098 ELSE
1099 string = vstring(is:ie)
1100 cval(nval) = trim(adjustl(string))
1101 END IF
1102 isstring = .true.
1103 END IF
1104 is = iblank + 1
1105 ie = lval
1106 decflag = .false.
1107 END IF
1108 END DO
1109 END SELECT ! keyWord
1110 END IF ! kExtract
1111 status = nval
1112 END IF
1113
1114 mystatus = status
1115
1116 RETURN
1117
1118 END FUNCTION parseline
