69 SUBROUTINE readmesh()
70
71 USE pahm_global, ONLY : meshfilenamespecified, meshfilename, meshfiletype, meshfileform
72 USE utilities, ONLY : touppercase
73
74 IMPLICIT NONE
75
76
77 CALL setmessagesource("ReadMesh")
78
79 IF (meshfilenamespecified .EQV. .false.) THEN
80 WRITE(scratchmessage, '(a)') 'ReadMesh: First specify a valid grid filename to proceed: ' // &
81 '[' // trim(meshfilename) // ']'
82 CALL allmessage(error, scratchmessage)
83 CALL terminate()
84 END IF
85
86 SELECT CASE(touppercase(meshfiletype))
87 !----- ADCIRC case
88 CASE('ADCIRC')
89 SELECT CASE(touppercase(meshfileform))
90 CASE('ASCII')
91 CALL readmeshasciifort14()
92
93 CASE('NETCDF')
94 WRITE(scratchmessage, '(a)') 'ReadMesh: NetCDF format is not supported yet for the mesh file type: ' // &
95 '[' // trim(meshfiletype) // ']'
96 CALL allmessage(error, scratchmessage)
97 CALL terminate()
98
99 CASE DEFAULT
100 WRITE(scratchmessage, '(a)') 'ReadMesh: Only ASCII and NetCDF formats are supported for the mesh file type: ' // &
101 '[' // trim(meshfiletype) // ']'
102 CALL allmessage(error, scratchmessage)
103 CALL terminate()
104 END SELECT
105
106 !----- SCHISM case
107 CASE('SCHISM')
108 SELECT CASE(touppercase(meshfileform))
109 CASE('ASCII')
110 CALL readmeshasciifort14()
111
112 CASE('NETCDF')
113 WRITE(scratchmessage, '(a)') 'ReadMesh: NetCDF format is not supported yet for the mesh file type: ' // &
114 '[' // trim(meshfiletype) // ']'
115 CALL allmessage(error, scratchmessage)
116 CALL terminate()
117
118 CASE DEFAULT
119 WRITE(scratchmessage, '(a)') 'ReadMesh: Only ASCII and NetCDF formats are supported for the mesh file type: ' // &
120 '[' // trim(meshfiletype) // ']'
121 CALL allmessage(error, scratchmessage)
122 CALL terminate()
123 END SELECT
124
125 !----- FVCOM case
126 CASE('FVCOM')
127 WRITE(scratchmessage, '(a)') 'ReadMesh: This file type is not yet imlemented: ' // &
128 '[' // trim(meshfiletype) // ']'
129 CALL allmessage(error, scratchmessage)
130 CALL terminate()
131
132 !----- ROMS case
133 CASE('ROMS')
134 WRITE(scratchmessage, '(a)') 'ReadMesh: This file type is not yet imlemented: ' // &
135 '[' // trim(meshfiletype) // ']'
136 CALL allmessage(error, scratchmessage)
137 CALL terminate()
138
139 !----- GENERIC case
140 CASE('GENERIC')
141 WRITE(scratchmessage, '(a)') 'ReadMesh: This file type is not yet imlemented: ' // &
142 '[' // trim(meshfiletype) // ']'
143 CALL allmessage(error, scratchmessage)
144 CALL terminate()
145
146 CASE DEFAULT
147 WRITE(scratchmessage, '(a)') 'ReadMesh: Invalid mesh file type specified: ' // &

170 SUBROUTINE readmeshasciifort14()
171
172 USE pahm_global, ONLY : lun_inp, meshfilename
173 USE utilities !PV specify what are we using here from utilities
174
175 IMPLICIT NONE
176
177 INTEGER, PARAMETER :: iUnit = lun_inp ! LUN for read operations
178 INTEGER :: ios ! I/O status
179 CHARACTER(LEN=512) :: fmtStr ! String to hold formats for I/O
180 INTEGER :: lineNum ! Line number currently being read
181
182 INTEGER :: labNodes, numFNodes ! Label and number of nodal faces for that label
183 INTEGER :: iCnt ! Counters
184
185
186 CALL setmessagesource("ReadMeshASCIIFort14")
187
188 CALL openfileforread(iunit, trim(meshfilename), ios)
189
190 linenum = 1
191
192 READ(unit=iunit, fmt='(a80)', err=10, END=20, IOSTAT=ios) agrid
193 linenum = linenum + 1
194
195 CALL logmessage(info, "Reading the mesh file: " // trim(meshfilename))
196 CALL logmessage(info, "Mesh file comment line: " // trim(agrid))
197 CALL logmessage(info, "Reading mesh file dimensions and coordinates.")
198
199 READ(unit=iunit, fmt=*, err=10, END=20, IOSTAT=ios) ne, np
200 linenum = linenum + 1
201
202 CALL allocatenodalandelementalarrays()
203
204 ! N O D E T A B L E
205 DO icnt = 1, np
206 READ(unit=iunit, fmt=*, err=10, END=20, IOSTAT=ios) labNodes, slam(iCnt), sfea(iCnt), dp(iCnt)
207
208 ! Check for (invalid longitude, latitude) values.
209 ! Currently only geographical coordinates are supported.
210 IF (.NOT. ((slam(icnt) >= -180.0_sz) .AND. (slam(icnt) <= 180.0_sz)) .OR. &
211 .NOT. ((sfea(icnt) >= -90.0_sz) .AND. (sfea(icnt) <= 90.0_sz))) THEN
212
213 fmtstr = '("Input file: ' // trim(meshfilename) // '", ", line ", i0,'
214 fmtstr = trim(fmtstr) // ' " contains invalid (lon, lat) values: ", " [", f14.4, ", ", f14.4, "]" '
215 fmtstr = trim(fmtstr) // ' " (should be degrees east and degrees north)")'
216 WRITE(scratchmessage, trim(fmtstr)) linenum, slam(icnt), sfea(icnt)
217
218 CALL allmessage(error, scratchmessage)
219 CLOSE(iunit)
220 CALL terminate()
221 END IF
222
223 linenum = linenum + 1
224 END DO
225
226 ! E L E M E N T T A B L E
227 DO icnt = 1, ne
228 READ(unit=iunit, fmt=*, err=10, END=20, IOSTAT=ios) labNodes, numfnodes
229
230 ! Check if numFNodes in the line is beyond the value of parameter MAXFACENODES,
231 ! to avoid out of bounds errors for the array "nm".
232 IF (numfnodes > maxfacenodes) THEN
233 fmtstr = '("Input file: ' // trim(meshfilename) // '", ", reading line ", i0,'
234 fmtstr = trim(fmtstr) // ' " gave a number of face nodes equal to: ", i0, '
235 fmtstr = trim(fmtstr) // ' ", which is greater than MAXFACENODES")'
236 WRITE(scratchmessage, trim(fmtstr)) linenum, numfnodes
237
238 CALL allmessage(error, scratchmessage)
239 CLOSE(iunit)
240 CALL terminate()
241 ELSE
242 backspace(unit=iunit)
243 END IF
244
245 READ(unit=iunit, fmt=*, err=10, END=20, IOSTAT=ios) labNodes, nfn(iCnt), nm(iCnt, 1:nfn(iCnt))
246
247 linenum = linenum + 1
248 END DO
249
250 CLOSE(iunit)
251
252 !PV Need to also check if arrays contain any missing values
253 IF ((comparereals(slam0, rmissv) == 0) .OR. &
254 (comparereals(sfea0, rmissv) == 0)) THEN
255 slam0 = sum(slam, 1) / np
256 sfea0 = sum(sfea, 1) / np
257 END IF
258
259 CALL geotocpp(sfea, slam, sfea0, slam0, xcslam, ycsfea)
260
261 CALL logmessage(info, 'Finished reading mesh file dimensions and coordinates.')
262
263 CALL unsetmessagesource()
264
265 ismeshok = .true.
266
267 RETURN
268
269 ! Jump to here on error condition during read
270 10 fmtstr = '("Reading line ", i0, " gave the following error code: ", i0, ".")'
271 WRITE(scratchmessage, fmtstr) linenum, ios
272
273 CALL allmessage(error, scratchmessage)
274 CLOSE(iunit)
275 CALL terminate()
276
277 ! Jump to here on end condition during read
278 20 fmtstr = '("Reached premature end of file on line ", i0, ".")'
279 WRITE(scratchmessage, trim(fmtstr)) linenum
280
281 CALL allmessage(error, scratchmessage)
282 CLOSE(iunit)
283 CALL terminate()
284
285 END SUBROUTINE readmeshasciifort14
