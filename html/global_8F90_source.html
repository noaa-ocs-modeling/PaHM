250 REAL(sz) function airdensity(atmt, atmp, relhum) result(myvalout)
251
252 IMPLICIT NONE
253
254 REAL(sz), INTENT(IN) :: atmt ! Surface temperature in degrees C (-50.0 <= T <= 100.0)
255 REAL(sz), INTENT(IN) :: atmp ! Atmospheric pressure (mb)
256 REAL(sz), INTENT(IN) :: relhum ! Relative humidity (0 - 100)
257
258 ! Local variables
259 REAL(hp) :: es, p, pv, pd, rh
260 REAL(hp) :: rd, rv, temp, tempk, dens
261
262
263 rh = relhum
264 IF (rh < 0.01) rh = 0.01_hp
265 IF (rh > 100.0) rh = 100.0_hp
266
267 temp = atmt
268 IF (temp < -50.0) temp = -50.0_hp
269 IF (temp > 100.0) temp = 100.0_hp
270
271 rd = 287.058_hp ! specific gas constant for dry air (J/kg*K)
272 rv = 461.495_hp ! specific gas constant for water vapor (J/kg*K)
273
274 ! Convert relative humidity to %
275 rh = 0.01_sz * rh
276
277 ! Convert atmT (C) to K
278 tempk = temp + 273.15_hp
279
280 ! Calculate the saturated vapor pressure (mb)
281 ! Temperature is in degrees Celcius
282 p = 0.99999683e+00_hp + temp * (-0.90826951e-02_hp + temp * (0.78736169e-04_hp + temp * &
283 (-0.61117958e-06_hp + temp * (0.43884187e-08_hp + temp * &
284 (-0.29883885e-10_hp + temp * (0.21874425e-12_hp + temp * &
285 (-0.17892321e-14_hp + temp * (0.11112018e-16_hp + temp * &
286 (-0.30994571e-19_hp)))))))))
287 es = 6.1078_hp / p**8 ! saturated vapour pressure (mb)
288
289 ! Calculate the actual vapor pressure (mb)
290 pv = es * rh
291
292 ! Calculate the actual vapor pressure
293 pd = atmp - pv
294
295 ! Convert the pressures from mb to Pa
296 pd = pd * 100.0_hp
297 pv = pv * 100.0_hp
298
299 ! Calculate the air density
300 dens = pd / (rd * tempk) + pv / (rv * tempk)
301
302 myvalout = dens
303
304 RETURN
305
306 END FUNCTION airdensity
