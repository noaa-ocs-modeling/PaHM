108 SUBROUTINE calcintensitychange(var, times, calcInt, status, order)
109
110 USE pahm_global, ONLY : deg2rad
111 USE utilities, ONLY : sphericaldistance
112
113 IMPLICIT NONE
114
115 REAL(sz), DIMENSION(:), INTENT(IN) :: var, times
116 INTEGER, OPTIONAL, INTENT(IN) :: order
117
118 REAL(sz), DIMENSION(:), INTENT(OUT) :: calcint
119 INTEGER, INTENT(OUT) :: status
120
121 INTEGER :: ordacur
122 REAL(sz) :: dt1, dt2
123 LOGICAL :: dt1ok, dt2ok
124 REAL(sz) :: val1, val2
125 INTEGER :: icnt, maxcnt
126
127 status = 0
128 maxcnt = 0
129
130 CALL setmessagesource("CalcIntensityChange")
131
132 IF ((SIZE(shape(var)) /= 1) .OR. (SIZE(shape(times)) /= 1)) THEN
133 WRITE(scratchmessage, '(a)') 'The rank of arrays var and times should be equal to 1 (vectors)'
134 CALL allmessage(error, scratchmessage)
135
136 CALL unsetmessagesource()
137
138 status = 1
139
140 RETURN
141 ELSE
142 maxcnt = SIZE(var)
143 END IF
144
145 ordacur = 2
146 IF (PRESENT(order)) THEN
147 IF (order <= 1) ordacur = 1
148 IF (order > 1) ordacur = 2
149 END IF
150 IF (SIZE(var) < 3) ordacur = 1
151
152 ! Case 1st orded accuracy using backward differences
153 IF (ordacur == 1 )THEN
154 DO icnt = 2, maxcnt
155 dt1 = times(icnt) - times(icnt - 1)
156 dt1ok = (comparereals(dt1, 0.0_sz) /=0)
157
158 val1 = 0.0_sz
159 IF (dt1ok) val1 = (var(icnt) - var(icnt - 1)) / dt1
160
161 calcint(icnt) = val1
162 END DO
163 calcint(1) = calcint(2)
164
165 CALL unsetmessagesource()
166
167 RETURN
168 END IF
169
170 ! Case 2nd order accuracy using Forward differences for the first point,
171 ! backward differences for the last point and central differences in
172 ! between points. Temporal spacing assumed to be uneven (general case).
173 ! Forward, backward and central differences are all 2nd order accurate
174 ! approximations.
175
176 !----- Forward differences (first point)
177 icnt = 1
178 dt1 = times(icnt + 1) - times(icnt)
179 dt1ok = (comparereals(dt1, 0.0_sz) /=0)
180 dt2 = times(icnt + 2) - times(icnt + 1)
181 dt2ok = (comparereals(dt2, 0.0_sz) /=0)
182
183 val1 = 0.0_sz
184 IF (dt1ok) val1 = (var(icnt + 1) - var(icnt)) / dt1
185
186 val2 = 0.0_sz
187 IF (dt2ok) val2 = (var(icnt + 2) - var(icnt + 1)) / dt2
188
189 IF (dt1ok .AND. dt2ok) THEN
190 calcint(icnt) = ((2.0_sz * dt1 + dt2) / (dt1 + dt2)) * val1 - (dt1 / (dt1 + dt2)) * val2
191 ELSE IF (.NOT. dt1ok) THEN
192 calcint(icnt) = val1
193 ELSE
194 calcint(icnt) = 2.0_sz * val1 - val2
195 END IF
196 !----- Forward differences (first point)
197
198 !----- Central differences
199 DO icnt = 2, maxcnt - 1
200 ! Forward
201 dt1 = times(icnt + 1) - times(icnt)
202 dt1ok = (comparereals(dt1, 0.0_sz) /=0)
203 ! Backward
204 dt2 = times(icnt) - times(icnt - 1)
205 dt2ok = (comparereals(dt2, 0.0_sz) /=0)
206
207 val1 = 0.0_sz
208 IF (dt1ok) val1 = (var(icnt + 1) - var(icnt)) / dt1
209
210 val2 = 0.0_sz
211 IF (dt2ok) val2 = (var(icnt) - var(icnt - 1)) / dt2
212
213 IF (dt1ok .AND. dt2ok) THEN
214 calcint(icnt) = (dt2 / (dt1 + dt2)) * val1 + (dt1 / (dt1 + dt2)) * val2
215 ELSE IF (.NOT. dt1ok) THEN
216 calcint(icnt) = val1
217 ELSE
218 calcint(icnt) = val2
219 END IF
220 END DO
221 !----- Central differences
222
223 !----- Backward differences (last point)
224 icnt = maxcnt
225 dt1 = times(icnt) - times(icnt - 1)
226 dt1ok = (comparereals(dt1, 0.0_sz) /=0)
227 dt2 = times(icnt - 1) - times(icnt - 2)
228 dt2ok = (comparereals(dt2, 0.0_sz) /=0)
229
230 val1 = 0.0_sz
231 IF (dt1ok) val1 = (var(icnt) - var(icnt - 1)) / dt1
232
233 val2 = 0.0_sz
234 IF (dt2ok) val2 = (var(icnt - 1) - var(icnt - 2)) / dt2
235
236 IF (dt1ok .AND. dt2ok) THEN
237 calcint(icnt) = ((2.0_sz * dt1 + dt2) / (dt1 + dt2)) * val1 - (dt1 / (dt1 + dt2)) * val2
238 ELSE IF (.NOT. dt1ok) THEN
239 calcint(icnt) = val1
240 ELSE
241 calcint(icnt) = 2.0_sz * val1 - val2
242 END IF
243 !----- Backward differences (last point)
244
245 CALL unsetmessagesource()
246
247 END SUBROUTINE calcintensitychange

282 SUBROUTINE uvtrans(lat, lon, times, u, v, status, order)
283
284 USE pahm_global, ONLY : deg2rad
285 USE utilities, ONLY : sphericaldistance
286
287 IMPLICIT NONE
288
289 REAL(sz), DIMENSION(:), INTENT(IN) :: lat, lon, times
290 INTEGER, OPTIONAL, INTENT(IN) :: order
291
292 REAL(sz), DIMENSION(:), INTENT(OUT) :: u, v
293 INTEGER, INTENT(OUT) :: status
294
295 INTEGER :: ordacur
296 REAL(sz) :: dx1, dy1, dx2, dy2
297 REAL(sz) :: dt1, dt2
298 LOGICAL :: dt1ok, dt2ok
299 REAL(sz) :: u1, u2, v1, v2
300 INTEGER :: icnt, maxcnt
301
302 status = 0
303 maxcnt = 0
304
305 CALL setmessagesource("UVTrans")
306
307 IF ((SIZE(shape(lat)) /= 1) .OR. (SIZE(shape(lon)) /= 1) .OR. (SIZE(shape(times)) /= 1)) THEN
308 WRITE(scratchmessage, '(a)') 'The rank of arrays lat, lon and times should be equal to 1 (vectors)'
309 CALL allmessage(error, scratchmessage)
310
311 CALL unsetmessagesource()
312
313 status = 1
314
315 RETURN
316 ELSE
317 maxcnt = SIZE(lat)
318 END IF
319
320 ordacur = 2
321 IF (PRESENT(order)) THEN
322 IF (order <= 1) ordacur = 1
323 IF (order > 1) ordacur = 2
324 END IF
325 IF (SIZE(lat) < 3) ordacur = 1
326
327 ! Case 1st orded accuracy using backward differences
328 IF (ordacur == 1 )THEN
329 DO icnt = 2, maxcnt
330 dx1 = sphericaldistance(lat(icnt - 1), lon(icnt - 1), lat(icnt - 1), lon(icnt))
331 dy1 = sphericaldistance(lat(icnt - 1), lon(icnt - 1), lat(icnt), lon(icnt - 1))
332 dt1 = abs(times(icnt) - times(icnt - 1))
333 dt1ok = (comparereals(dt1, 0.0_sz) /=0)
334
335 u1 = 0.0_sz
336 v1 = 0.0_sz
337 IF (dt1ok) THEN
338 u1 = sign(dx1 / dt1, (lon(icnt) - lon(icnt - 1)))
339 v1 = sign(dy1 / dt1, (lat(icnt) - lat(icnt - 1)))
340 END IF
341
342 u(icnt) = u1
343 v(icnt) = v1
344 END DO
345 u(1) = u(2)
346 v(1) = v(2)
347
348 CALL unsetmessagesource()
349
350 RETURN
351 END IF
352
353 ! Case 2nd order accuracy using Forward differences for the first point,
354 ! backward differences for the last point and central differences in
355 ! between points. Temporal spacing assumed to be uneven (general case).
356 ! Forward, backward and central differences are all 2nd order accurate
357 ! approximations.
358
359 !----- Forward differences (first point)
360 icnt = 1
361 dx1 = sphericaldistance(lat(icnt), lon(icnt), lat(icnt), lon(icnt + 1))
362 dy1 = sphericaldistance(lat(icnt), lon(icnt), lat(icnt + 1), lon(icnt))
363 dt1 = abs(times(icnt + 1) - times(icnt))
364 dt1ok = (comparereals(dt1, 0.0_sz) /=0)
365
366 dx2 = sphericaldistance(lat(icnt + 1), lon(icnt + 1), lat(icnt + 1), lon(icnt + 2))
367 dy2 = sphericaldistance(lat(icnt + 1), lon(icnt + 1), lat(icnt + 2), lon(icnt + 1))
368 dt2 = abs(times(icnt + 2) - times(icnt + 1))
369 dt2ok = (comparereals(dt2, 0.0_sz) /=0)
370
371 u1 = 0.0_sz
372 v1 = 0.0_sz
373 IF (dt1ok) THEN
374 u1 = sign(dx1 / dt1, (lon(icnt + 1) - lon(icnt)))
375 v1 = sign(dy1 / dt1, (lat(icnt + 1) - lat(icnt)))
376 END IF
377
378 u2 = 0.0_sz
379 v2 = 0.0_sz
380 IF (dt2ok) THEN
381 u2 = sign(dx2 / dt2, (lon(icnt + 2) - lon(icnt + 1)))
382 v2 = sign(dy2 / dt2, (lat(icnt + 2) - lat(icnt + 1)))
383 END IF
384
385 IF (dt1ok .AND. dt2ok) THEN
386 u(icnt) = ((2.0_sz * dt1 + dt2) / (dt1 + dt2)) * u1 - (dt1 / (dt1 + dt2)) * u2
387 v(icnt) = ((2.0_sz * dt1 + dt2) / (dt1 + dt2)) * v1 - (dt1 / (dt1 + dt2)) * v2
388 ELSE IF (.NOT. dt1ok) THEN
389 u(icnt) = u1
390 v(icnt) = v1
391 ELSE
392 u(icnt) = 2.0_sz * u1 - u2
393 v(icnt) = 2.0_sz * v1 - v2
394 END IF
395 !----- Forward differences (first point)
396
397 !----- Central differences
398 DO icnt = 2, maxcnt - 1
399 ! Forward
400 dx1 = sphericaldistance(lat(icnt), lon(icnt), lat(icnt), lon(icnt + 1))
401 dy1 = sphericaldistance(lat(icnt), lon(icnt), lat(icnt + 1), lon(icnt))
402 dt1 = abs(times(icnt + 1) - times(icnt))
403 dt1ok = (comparereals(dt1, 0.0_sz) /=0)
404 ! Backward
405 dx2 = sphericaldistance(lat(icnt - 1), lon(icnt - 1), lat(icnt - 1), lon(icnt))
406 dy2 = sphericaldistance(lat(icnt - 1), lon(icnt - 1), lat(icnt), lon(icnt - 1))
407 dt2 = abs(times(icnt) - times(icnt - 1))
408 dt2ok = (comparereals(dt2, 0.0_sz) /=0)
409
410 u1 = 0.0_sz
411 v1 = 0.0_sz
412 IF (dt1ok) THEN
413 u1 = sign(dx1 / dt1, (lon(icnt + 1) - lon(icnt)))
414 v1 = sign(dy1 / dt1, (lat(icnt + 1) - lat(icnt)))
415 END IF
416
417 u2 = 0.0_sz
418 v2 = 0.0_sz
419 IF (dt2ok) THEN
420 u2 = sign(dx2 / dt2, (lon(icnt) - lon(icnt - 1)))
421 v2 = sign(dy2 / dt2, (lat(icnt) - lat(icnt - 1)))
422 END IF
423
424 IF (dt1ok .AND. dt2ok) THEN
425 u(icnt) = (dt2 / (dt1 + dt2)) * u1 + (dt1 / (dt1 + dt2)) * u2
426 v(icnt) = (dt2 / (dt1 + dt2)) * v1 + (dt1 / (dt1 + dt2)) * v2
427 ELSE IF (.NOT. dt1ok) THEN
428 u(icnt) = u1
429 v(icnt) = v1
430 ELSE
431 u(icnt) = u2
432 v(icnt) = v2
433 END IF
434 END DO
435 !----- Central differences
436
437 !----- Backward differences (last point)
438 icnt = maxcnt
439 dx1 = sphericaldistance(lat(icnt), lon(icnt), lat(icnt), lon(icnt - 1))
440 dy1 = sphericaldistance(lat(icnt), lon(icnt), lat(icnt - 1), lon(icnt))
441 dt1 = abs(times(icnt) - times(icnt - 1))
442 dt1ok = (comparereals(dt1, 0.0_sz) /=0)
443
444 dx2 = sphericaldistance(lat(icnt - 1), lon(icnt - 1), lat(icnt - 1), lon(icnt - 2))
445 dy2 = sphericaldistance(lat(icnt - 1), lon(icnt - 1), lat(icnt - 2), lon(icnt - 1))
446 dt2 = abs(times(icnt - 1) - times(icnt - 2))
447 dt2ok = (comparereals(dt2, 0.0_sz) /=0)
448
449 u1 = 0.0_sz
450 v1 = 0.0_sz
451 IF (dt1ok) THEN
452 u1 = sign(dx1 / dt1, (lon(icnt) - lon(icnt - 1)))
453 v1 = sign(dy1 / dt1, (lat(icnt) - lat(icnt - 1)))
454 END IF
455
456 u2 = 0.0_sz
457 v2 = 0.0_sz
458 IF (dt2ok) THEN
459 u2 = sign(dx2 / dt2, (lon(icnt - 1) - lon(icnt - 2)))
460 v2 = sign(dy2 / dt2, (lat(icnt - 1) - lat(icnt - 2)))
461 END IF
462
463 IF (dt1ok .AND. dt2ok) THEN
464 u(icnt) = ((2.0_sz * dt1 + dt2) / (dt1 + dt2)) * u1 - (dt1 / (dt1 + dt2)) * u2
465 v(icnt) = ((2.0_sz * dt1 + dt2) / (dt1 + dt2)) * v1 - (dt1 / (dt1 + dt2)) * v2
466 ELSE IF (.NOT. dt1ok) THEN
467 u(icnt) = u1
468 v(icnt) = v1
469 ELSE
470 u(icnt) = 2.0_sz * u1 - u2
471 v(icnt) = 2.0_sz * v1 - v2
472 END IF
473 !----- Backward differences (last point)
474
475 CALL unsetmessagesource()
476
477 END SUBROUTINE uvtrans

742 !----------------------------------------------------------------
743 SUBROUTINE calcrmaxes()
744
745 IMPLICIT NONE
746
747 REAL(sz) :: root ! Radius of maximum winds
748 REAL(sz), PARAMETER :: innerradius = 1.0_sz
749 REAL(sz), PARAMETER :: outerradius = 400.0_sz
750 REAL(sz), PARAMETER :: accuracy = 0.0001_sz
751 REAL(sz), PARAMETER :: zoom = 0.01_sz
752 INTEGER , PARAMETER :: itermax = 3
753 REAL(sz) :: r1, r2, r3, r4, dr
754 REAL(sz) :: vicinity
755 INTEGER :: n, iter
756
757 !-----------------------------
758 ! Loop over quadrants of storm
759 !-----------------------------
760 DO n = 1, nquads
761 ! set B and vMax values for each quadrant
762 ! for nws19, B and vMax are constant
763 ! for simplified nws20, B is constant, while vMax is not
764 b = bs(n + 1)
765 vmax = vmbl(n + 1)
766
767 quad = n
768 root = -1.0_sz
769 r1 = innerradius
770 r2 = outerradius
771 dr = 1.0_sz
772 DO iter = 1, itermax
773 root = findroot(vhwithcori, r1, r2, dr, r3, r4)
774 r1 = r3
775 r2 = r4
776 dr = dr * zoom
777 END DO
778
779 ! determine if rMax is actually in the vicinity of the
780 ! isotach radius that we are using to solve for rMax,
781 ! and if so, take another shot at finding the
782 ! rMax using the gradient wind balance that neglects
783 ! coriolis (and is appropriate in the vicinity of rMax)
784 !
785 ! CompareReals(r1, r2)
786 ! -1 (if r1 < r2)
787 ! 0 (if r1 = r2)
788 ! +1 (if r1 > r2)
789 vicinity = abs(root - radius(quad)) / root
790 !PV DEL IF ((root < 0.0_SZ) .OR. (vicinity <= 0.1_SZ)) THEN
791 IF (comparereals(root, 0.0_sz) == -1 .OR. comparereals(vicinity, 0.1_sz) /= 1) THEN
792 r1 = innerradius
793 r2 = outerradius
794 dr = 1.0_sz

1440 !----------------------------------------------------------------
1441 SUBROUTINE uvp(lat, lon, uTrans, vTrans, u, v, p)
1442
1443 USE pahm_global, ONLY : windreduction, one2ten, deg2rad, rad2deg, mb2pa, kt2ms, nm2m, m2nm, rearth
1444
1445 IMPLICIT NONE
1446
1447 REAL(sz), INTENT(IN) :: lat
1448 REAL(sz), INTENT(IN) :: lon
1449 REAL(sz), INTENT(IN) :: utrans
1450 REAL(sz), INTENT(IN) :: vtrans
1451
1452 REAL(sz), INTENT(OUT) :: u
1453 REAL(sz), INTENT(OUT) :: v
1454 REAL(sz), INTENT(OUT) :: p
1455
1456 REAL(sz) :: transspdx !NWS8-style translation speed
1457 REAL(sz) :: transspdy !NWS8-style translation speed
1458
1459 REAL(sz) :: dx
1460 REAL(sz) :: dy
1461 REAL(sz) :: dist
1462 REAL(sz) :: rmx
1463 REAL(sz) :: angle
1464 REAL(sz) :: speed
1465 REAL(sz) :: uf
1466 REAL(sz) :: vf
1467 REAL(sz) :: percentcoriolis
1468 REAL(sz) :: speedatrmax
1469 REAL(sz) :: vmaxfactor
1470
1471 !------------------------------------------------------
1472 ! Calculate distance and angle between eye of hurricane
1473 ! and input nodal point
1474 !------------------------------------------------------
1475 dx = deg2rad * rearth * (lon - clon) * cos(deg2rad * clat)
1476 dy = deg2rad * rearth * (lat - clat)
1477 dist = sqrt(dx * dx + dy * dy)
1478
1479 !----------------------------------------
1480 ! Handle special case at eye of hurricane
1481 ! in eye velocity is zero not translational velocity
1482 !----------------------------------------
1483 IF (dist < 1.0_sz) THEN
1484 u = 0.0_sz
1485 v = 0.0_sz
1486 p = pc * mb2pa
1487
1488 RETURN
1489 END IF
1490
1491 dist = m2nm * dist
1492
1493 angle = 360.0_sz + rad2deg * atan2(dx, dy)
1494 IF (angle > 360.0_sz) angle = angle - 360.0_sz
1495
1496 latestangle = angle
1497 rmx = rmw(angle)
1498 latestrmax = rmx
1499
1500 !---------------------------------------------------
1501 ! Compute (u,v) wind velocity components from the
1502 ! asymmetric hurricane vortex.
1503 !
1504 ! Note: the vortex winds are valid at the top of the
1505 ! surface layer, so reduce the winds to the surface.
1506 ! Also convert the winds from max sustained 1-minute
1507 ! averages to 10-minute averages for the storm surge
1508 ! model.
1509 !---------------------------------------------------
1510 percentcoriolis = 1.0_sz
1511 speed = sqrt((vmax * kt2ms)**2 * (rmx / dist)**b * exp(1.0_sz - (rmx / dist)**b) + &
1512 (nm2m * dist * percentcoriolis * corio / 2.0_sz)**2) &
1513 - nm2m * dist * percentcoriolis * corio / 2.0_sz
1514
1515 ! Calculate the wind speed (m/s) at rMax, using
1516 ! equation that includes full coriolis
1517 speedatrmax = sqrt((vmax * kt2ms)**2 * exp(0.0_sz) + &
1518 (nm2m * dist * percentcoriolis * corio / 2.0_sz)**2) &
1519 - nm2m * dist * percentcoriolis * corio / 2.0_sz
1520
1521 ! Calculate a factor to place the velocity profile so that
1522 ! it hits vMax
1523 vmaxfactor = vmax * kt2ms / speedatrmax
1524
1525 ! Calculate NWS8-like translation speed
1526 transspdx = (abs(speed / speedatrmax)) * utrans * kt2ms
1527 transspdy = (abs(speed / speedatrmax)) * vtrans * kt2ms
1528
1529 speed = speed * vmaxfactor
1530
1531 ! Now reduce the wind speed to the surface
1532 speed = speed * windreduction
1533
1534 u = -speed * cos(deg2rad * angle)
1535 v = speed * sin(deg2rad * angle)
