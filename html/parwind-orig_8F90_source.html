174 CHARACTER(LEN=512) :: inpLine, line
175 CHARACTER(LEN=512) :: fmtStr
176
177 INTEGER :: lenLine
178 INTEGER :: nLines ! Number of lines counter
179 INTEGER :: iFile, iCnt ! loop counters
180 INTEGER :: iUnit, errIO, ios, status
181
182 CHARACTER(LEN=10), ALLOCATABLE :: chkArrStr(:)

190 errio = 0

192 fmtstr = '(a2, 2x, i2, 2x, a10, 2x, i2, 2x, a4, 2x, i3, 2x, i3, a1, 2x, i4, a1, 2x, i3, 2x, i4, 2x, a2, '

196

198
199 ! Allocate the best track structure array. This structure holds all the
200 ! input values for the storm track as read in from the track input file
201 ! (a-deck, b-deck ATCF format) as well as the converted best track variables
202 ! (as appropriate).
203 ALLOCATE(besttrackdata(nbtrfiles))
204
205 ! This is the main loop. We loop through all the best track files
206 ! (user input)
207 DO ifile = 1, nbtrfiles
208 inpfile = besttrackfilename(ifile)
209
210 CALL openfileforread(iunit, trim(adjustl(inpfile)), errio)
211
212 IF (errio /= 0) THEN
213 WRITE(scratchmessage, '(a)') 'Error opening the best track file: ' // trim(adjustl(inpfile))
214 CALL allmessage(error, scratchmessage)
215
216 CALL unsetmessagesource()
217
218 CALL terminate()
219 ELSE
220 WRITE(scratchmessage, '(a)') 'Processing the best track file: ' // trim(adjustl(inpfile))
221 CALL logmessage(info, scratchmessage)
222 END IF
223
224 besttrackdata(ifile)%fileName = trim(adjustl(inpfile))
225 besttrackdata(ifile)%thisStorm = ""
226 besttrackdata(ifile)%loaded = .false.
227 besttrackdata(ifile)%numRec = -1
228
229 ! Count the number of non-empty or commented out lines in the file.
230 ! Comments are are considered those lines with the first non-blank character of "!" or "#"
231 nlines = 0
232 DO
233 READ(unit=iunit, fmt='(a)', err=10, END=5, IOSTAT=errIO) inpline
234
235 lenline = getlinerecord(inpline, line)
236 IF (lenline /= 0) nlines = nlines + 1
237 END DO
238 5 rewind(unit=iunit)
239
240 ! Array allocation in the structure bestTrackData
241 CALL allocbtrstruct(besttrackdata(ifile), nlines)
242
243 icnt = 0
244 DO WHILE (.true.)
245 READ(unit=iunit, fmt='(a)', err=10, END=20, IOSTAT=errIO) inpline
246
247 lenline = getlinerecord(inpline, line)
248
249 IF (lenline /= 0) THEN
250 icnt = icnt + 1
251 READ(line, fmt=fmtstr, err=11, iostat=ios) &
252 besttrackdata(ifile)%basin(icnt), besttrackdata(ifile)%cyNum(icnt), &
253 besttrackdata(ifile)%dtg(icnt), besttrackdata(ifile)%techNum(icnt), &
254 besttrackdata(ifile)%tech(icnt), besttrackdata(ifile)%tau(icnt), &
255 besttrackdata(ifile)%intLat(icnt), besttrackdata(ifile)%ns(icnt), &
256 besttrackdata(ifile)%intLon(icnt), besttrackdata(ifile)%ew(icnt), &
257 besttrackdata(ifile)%intVMax(icnt), besttrackdata(ifile)%intMslp(icnt), &
258 besttrackdata(ifile)%ty(icnt), besttrackdata(ifile)%rad(icnt), &
259 besttrackdata(ifile)%windCode(icnt), besttrackdata(ifile)%intRad1(icnt), &
260 besttrackdata(ifile)%intRad2(icnt), besttrackdata(ifile)%intRad3(icnt), &
261 besttrackdata(ifile)%intRad4(icnt), besttrackdata(ifile)%intPOuter(icnt), &
262 besttrackdata(ifile)%intROuter(icnt), besttrackdata(ifile)%intRmw(icnt), &
263 besttrackdata(ifile)%gusts(icnt), besttrackdata(ifile)%eye(icnt), &
264 besttrackdata(ifile)%subregion(icnt), besttrackdata(ifile)%maxseas(icnt), &
265 besttrackdata(ifile)%initials(icnt), besttrackdata(ifile)%dir(icnt), &
266 besttrackdata(ifile)%intSpeed(icnt), besttrackdata(ifile)%stormName(icnt), &
267 besttrackdata(ifile)%cycleNum(icnt)
268
269 !---------- Convert lat/lon values to S/N and W/E notations
270 IF (touppercase(besttrackdata(ifile)%ns(icnt)) == 'S') THEN
271 besttrackdata(ifile)%lat(icnt) = -0.1_sz * besttrackdata(ifile)%intLat(icnt)
272 ELSE
273 besttrackdata(ifile)%lat(icnt) = 0.1_sz * besttrackdata(ifile)%intLat(icnt)
274 END IF
275
276 IF (touppercase(besttrackdata(ifile)%ew(icnt)) == 'W') THEN
277 besttrackdata(ifile)%lon(icnt) = -0.1_sz * besttrackdata(ifile)%intLon(icnt)
278 ELSE
279 besttrackdata(ifile)%lon(icnt) = 0.1_sz * besttrackdata(ifile)%intLon(icnt)
280 END IF
281 !----------
282
283 !---------- Get the year, month, day, hour from the DGT string
284 READ(besttrackdata(ifile)%dtg(icnt)(1:4), fmt='(i4.4)', iostat=ios) besttrackdata(ifile)%year(icnt)
285 IF (ios /= 0) besttrackdata(ifile)%year(icnt) = -1
286 READ(besttrackdata(ifile)%dtg(icnt)(5:6), fmt='(i2.2)', iostat=ios) besttrackdata(ifile)%month(icnt)
287 IF (ios /= 0) besttrackdata(ifile)%month(icnt) = -1
288 READ(besttrackdata(ifile)%dtg(icnt)(7:8), fmt='(i2.2)', iostat=ios) besttrackdata(ifile)%day(icnt)
289 IF (ios /= 0) besttrackdata(ifile)%day(icnt) = -1
290 READ(besttrackdata(ifile)%dtg(icnt)(9:10), fmt='(i2.2)', iostat=ios) besttrackdata(ifile)%hour(icnt)
291 IF (ios /= 0) besttrackdata(ifile)%hour(icnt) = -1
292 !----------
293 END IF
294 END DO
295
296 10 IF (errio /= 0) THEN
297 WRITE(scratchmessage, '(a)') 'Error in file: ' // trim(adjustl(inpfile)) // &
298 ', while processing line: ' // trim(adjustl(inpline))
299 CALL allmessage(error, scratchmessage)
300
301 CLOSE(iunit)
302
303 CALL unsetmessagesource()
304
305 CALL terminate()
306 END IF
307
308 11 IF (ios /= 0) THEN
309 WRITE(scratchmessage, '(a)') 'Error in file: ' // trim(adjustl(inpfile)) // &
310 ', while processing line: ' // trim(adjustl(line))
311 CALL allmessage(error, scratchmessage)
312
313 CLOSE(iunit)
314
315 CALL unsetmessagesource()
316
317 CALL terminate()
318 END IF
319
320 20 CLOSE(iunit)
321
322 besttrackdata(ifile)%thisStorm = ''
323 besttrackdata(ifile)%loaded = .true.
324 besttrackdata(ifile)%numRec = nlines
325
326
327 !------------------------------------------------------------
328 ! Get the unique storm name and store it in the thisStorm string
329 ALLOCATE(chkarrstr(nlines))
330 ALLOCATE(idxarrstr(nlines))
331
332 nunique = charunique(besttrackdata(ifile)%stormName, chkarrstr, idxarrstr)
333
334 maxcnt = -1
335 DO kcnt = 1, nunique
336 kmax = count(chkarrstr(kcnt) == besttrackdata(ifile)%stormName)
337 IF (kmax > maxcnt) THEN
338 maxcnt = kmax
339 besttrackdata(ifile)%thisStorm = trim(adjustl(chkarrstr(kcnt)))
340 END IF
341 END DO
342
343 DEALLOCATE(chkarrstr)
344 DEALLOCATE(idxarrstr)
345 !------------------------------------------------------------
346
347 !------------------------------------------------------------
348 ! This is an extra step (paranoid) to ensure that the dates in the bestTrackData are
349 ! stored in ascending order
350 ALLOCATE(idx0(besttrackdata(ifile)%numRec))
351 ALLOCATE(idx1(besttrackdata(ifile)%numRec))
352
353 CALL indexx(besttrackdata(ifile)%dtg, idx1, status, .true.)
354
355 IF (status /= 0) THEN
356 CALL unsetmessagesource()
357
358 CALL terminate()
359 END IF
360
361 ! Create the index array to be used in the comparison below
362 idx0 = arth(1, 1, besttrackdata(ifile)%numRec)
363
364 IF (.NOT. arrayequal(idx0, idx1)) THEN
365 besttrackdata(ifile)%basin = besttrackdata(ifile)%basin(idx1)
366 besttrackdata(ifile)%cyNum = besttrackdata(ifile)%cyNum(idx1)
367 besttrackdata(ifile)%dtg = besttrackdata(ifile)%dtg(idx1)
368 besttrackdata(ifile)%techNum = besttrackdata(ifile)%techNum(idx1)
369 besttrackdata(ifile)%tech = besttrackdata(ifile)%tech(idx1)
370 besttrackdata(ifile)%tau = besttrackdata(ifile)%tau(idx1)
371 besttrackdata(ifile)%intLat = besttrackdata(ifile)%intLat(idx1)
372 besttrackdata(ifile)%ns = besttrackdata(ifile)%ns(idx1)
373 besttrackdata(ifile)%intLon = besttrackdata(ifile)%intLon(idx1)
374 besttrackdata(ifile)%ew = besttrackdata(ifile)%ew(idx1)
375 besttrackdata(ifile)%intVMax = besttrackdata(ifile)%intVMax(idx1)
376 besttrackdata(ifile)%intMslp = besttrackdata(ifile)%intMslp(idx1)
377 besttrackdata(ifile)%ty = besttrackdata(ifile)%ty(idx1)
378 besttrackdata(ifile)%rad = besttrackdata(ifile)%rad(idx1)
379 besttrackdata(ifile)%windCode = besttrackdata(ifile)%windCode(idx1)
380 besttrackdata(ifile)%intRad1 = besttrackdata(ifile)%intRad1(idx1)
381 besttrackdata(ifile)%intRad2 = besttrackdata(ifile)%intRad2(idx1)
382 besttrackdata(ifile)%intRad3 = besttrackdata(ifile)%intRad3(idx1)
383 besttrackdata(ifile)%intRad4 = besttrackdata(ifile)%intRad4(idx1)
384 besttrackdata(ifile)%intPOuter = besttrackdata(ifile)%intPOuter(idx1)
385 besttrackdata(ifile)%intROuter = besttrackdata(ifile)%intROuter(idx1)
386 besttrackdata(ifile)%intRmw = besttrackdata(ifile)%intRmw(idx1)
387 besttrackdata(ifile)%gusts = besttrackdata(ifile)%gusts(idx1)
388 besttrackdata(ifile)%eye = besttrackdata(ifile)%eye(idx1)
389 besttrackdata(ifile)%subregion = besttrackdata(ifile)%subregion(idx1)
390 besttrackdata(ifile)%maxseas = besttrackdata(ifile)%maxseas(idx1)
391 besttrackdata(ifile)%initials = besttrackdata(ifile)%initials(idx1)
392 besttrackdata(ifile)%dir = besttrackdata(ifile)%dir(idx1)
393 besttrackdata(ifile)%intSpeed = besttrackdata(ifile)%intSpeed(idx1)
394 besttrackdata(ifile)%stormName = besttrackdata(ifile)%stormName(idx1)
395 besttrackdata(ifile)%cycleNum = besttrackdata(ifile)%cycleNum(idx1)
396 END IF
397
398 DEALLOCATE(idx0)
399 DEALLOCATE(idx1)
400 !------------------------------------------------------------
401
402 CALL writebesttrackdata(besttrackfilename(ifile), besttrackdata(ifile), '_fort22fmt')
403 END DO ! End of "iFile" loop
404
405 CALL unsetmessagesource()
406
407 END SUBROUTINE readbesttrackfile

678 SUBROUTINE processhollanddata(idTrFile, strOut, status)
679
680 USE pahm_global, ONLY : nm2m, kt2ms, nbtrfiles
681 USE utilities, ONLY : touppercase, charunique
682 USE timedateutils, ONLY : timeconv
683 USE pahm_vortex, ONLY : calcintensitychange, uvtrans
684
685 IMPLICIT NONE
686
687 INTEGER, INTENT(IN) :: idTrFile
688 TYPE(hollanddata_t), INTENT(OUT) :: strOut
689 INTEGER, INTENT(OUT) :: status ! error status
690
691 ! numUniqRec, outDTG, idxDTG are used to identify the unique DTG elements in the input structure
692 INTEGER :: numUniqRec
693 CHARACTER(LEN=10), ALLOCATABLE :: outDTG(:)
694 INTEGER, ALLOCATABLE :: idxDTG(:)
695
696 INTEGER :: plIdx ! populated index for Holland Data array
697 INTEGER :: iCnt ! loop counters
698
699 CHARACTER(LEN=4) :: castType !hindcast,forecast
700 REAL(SZ), ALLOCATABLE :: castTime(:) ! seconds since start of year
701
702 REAL(SZ) :: spdVal, pressVal, rrpVal, rmwVal
703
704 status = 0 ! no error
705
706
707 CALL setmessagesource("ProcessHollandData")
708
709 IF ((idtrfile >= 1) .AND. (idtrfile <= nbtrfiles)) THEN
710 IF (.NOT. besttrackdata(idtrfile)%loaded) THEN
711 status = 2
712
713 WRITE(scratchmessage, '(a, i0)') 'Error while loading best track data structure with id: ', idtrfile
714 CALL allmessage(error, scratchmessage)
715
716 CALL unsetmessagesource()
717
718 RETURN
719 END IF
720 ELSE
721 status = 1
722
723 WRITE(scratchmessage, '(a, i0, a, i0)') 'Wrong best track structure id (idTrFile): ', idtrfile, &
724 ', it should be: (1<= idTrFile <= nBTrFiles); nBTrFiles = ', nbtrfiles
725 CALL allmessage(error, scratchmessage)
726
727 CALL unsetmessagesource()
728
729 RETURN
730 END IF
731
732 WRITE(scratchmessage, '(a, i0)') 'Processing the best track structure with id: ', idtrfile
733 CALL logmessage(info, scratchmessage)
734
735 ! Most likely the array size will be larger if repeated times are found
736 ! in the best track structure.
737 ALLOCATE(outdtg(besttrackdata(idtrfile)%numRec))
738 ALLOCATE(idxdtg(besttrackdata(idtrfile)%numRec))
739
740 ! Get unique lines that represent new points in time.
741 ! Repeated time points occur in hindcasts for the purpose of
742 ! describing winds in the quadrants of the storm. We don't use the
743 ! quadrant-by-quadrant wind data. Repeated time data occur in the
744 ! forecast because the time data is just the time that the forecast
745 ! was made. The important parameter in the forecast file is the
746 ! forecast increment.
747 numuniqrec = charunique(besttrackdata(idtrfile)%dtg, outdtg, idxdtg)

750 ! Populate the Holland structure
751 !--------------------
752 CALL allochollstruct(strout, numuniqrec)

754 ALLOCATE(casttime(numuniqrec))
755
756 strout%fileName = besttrackdata(idtrfile)%fileName
757 strout%thisStorm = besttrackdata(idtrfile)%thisStorm

765 plidx = idxdtg(icnt)
766
767 casttype = touppercase(trim(adjustl(besttrackdata(idtrfile)%tech(plidx))))
768
769 ! Convert speeds from knots to m/s
770 spdval = kt2ms * besttrackdata(idtrfile)%intVMax(plidx)
771
772 ! Convert pressure(s) from mbar to Pa
773 pressval = 100.0_sz * besttrackdata(idtrfile)%intMslp(plidx)
774
775 ! Convert all distances from nm to km/m
776 rrpval = nm2m * besttrackdata(idtrfile)%intROuter(plidx) ! in m

780 strout%stormNumber(icnt) = besttrackdata(idtrfile)%cyNum(plidx)
781 strout%dtg(icnt) = besttrackdata(idtrfile)%dtg(plidx)

783 strout%month(icnt) = besttrackdata(idtrfile)%month(plidx)
784 strout%day(icnt) = besttrackdata(idtrfile)%day(plidx)
785 strout%hour(icnt) = besttrackdata(idtrfile)%hour(plidx)
786 strout%castType(icnt) = besttrackdata(idtrfile)%tech(plidx)

788 strout%iLat(icnt) = besttrackdata(idtrfile)%intLat(plidx)
789 strout%lat(icnt) = besttrackdata(idtrfile)%lat(plidx)
790 strout%iLon(icnt) = besttrackdata(idtrfile)%intLon(plidx)
791 strout%lon(icnt) = besttrackdata(idtrfile)%lon(plidx)
792
793 strout%iSpeed(icnt) = besttrackdata(idtrfile)%intVMax(plidx)
794 strout%speed(icnt) = spdval
795 strout%iCPress(icnt) = besttrackdata(idtrfile)%intMslp(plidx)
796 strout%cPress(icnt) = pressval
797 strout%iRrp(icnt) = besttrackdata(idtrfile)%intROuter(plidx)
798 strout%rrp(icnt) = rrpval
799 strout%iRmw(icnt) = besttrackdata(idtrfile)%intRmw(plidx)
800 strout%rmw(icnt) = rmwval
801
802 ! PV check if this SELECT code is actually needed. Need to check the different format
803 ! of input files.
804 SELECT CASE(casttype)
805 CASE("BEST") ! nowcast/hindcast
806 ! PV check if this is needed
807 CALL timeconv(strout%year(icnt), strout%month(icnt), strout%day(icnt), strout%hour(icnt), 0, 0.0_sz, casttime(icnt))
808
809 CASE("OFCL") ! forecast
810 ! PV check if this is needed
811 IF (icnt > 1) THEN
812 IF ( (strout%fcstInc(icnt) /= 0) .AND. (strout%fcstInc(icnt) == strout%fcstInc(icnt - 1))) cycle
813 END IF
814
815 IF (strout%fcstInc(icnt) == 0) THEN
816 CALL timeconv(strout%year(icnt), strout%month(icnt), strout%day(icnt), &
817 strout%hour(icnt), 0, 0.0_sz, casttime(icnt))
818 ELSE
819 casttime(icnt) = casttime(icnt - 1) + (strout%fcstInc(icnt) - strout%fcstInc(icnt - 1) * 3600.0_sz)
820 END IF
821
822 IF ((strout%iCPress(icnt) == 0) .OR. (strout%iRmw(icnt) == 0)) THEN
823 CALL allmessage(error, &
824 'The storm hindcast/forecast input file ' // trim(strout%fileName) // &
825 ' contains invalid data for central pressure or rMax.')
826 CALL terminate()
827 END IF
828
829 ! Adding a new type to allow the analyst to add lines
830 ! that do nothing but produce zero winds and background barometric
831 ! pressure. These lines can have a date/time like a BEST line or
832 ! a date/time and forecast period like an OFCL line.
833 CASE("CALM")
834 ! PV check if this is needed
835 WRITE(scratchmessage, '(a)') 'The file: ' // trim(strout%fileName) // ' contains at least one "CALM" line.'
836 CALL logmessage(echo, scratchmessage)
837
838 IF (icnt > 1) THEN
839 IF ( (strout%fcstInc(icnt) /= 0) .AND. (strout%fcstInc(icnt) == strout%fcstInc(icnt - 1))) cycle
840 END IF
841
842 IF (strout%fcstInc(icnt) == 0) THEN
843 CALL timeconv(strout%year(icnt), strout%month(icnt), strout%day(icnt), &
844 strout%hour(icnt), 0, 0.0_sz, casttime(icnt))
845 ELSE
846 casttime(icnt) = casttime(icnt - 1) + (strout%fcstInc(icnt) - strout%fcstInc(icnt - 1) * 3600.0_sz)
847 END IF
848
849 CASE DEFAULT ! unrecognized
850 WRITE(scratchmessage, '(a)') 'Only "BEST", "OFCL", or "CALM" are allowed in the 5th column of ' // &
851 trim(adjustl(strout%fileName))
852 CALL allmessage(error, scratchmessage)
853
854 CALL terminate()
855 END SELECT
856
857 strout%castTime(icnt) = casttime(icnt)
858 END DO ! numUniqRec
859
860 ! Calculate the cPress intensity change (dP/dt)
861 CALL calcintensitychange(strout%cPress, casttime, strout%cPrDt, status, 2)
862
863 ! Calculate storm translation velocities based on change in position,
864 ! approximate u and v translation velocities
865 CALL uvtrans(strout%lat, strout%lon, casttime, strout%trVx, strout%trVy, status, 2)
866
867 DEALLOCATE(casttime)
868!--------------------
869
870 DEALLOCATE(outdtg)
871 DEALLOCATE(idxdtg)
872
873 CALL unsetmessagesource()
874
875 END SUBROUTINE processhollanddata

1046 ! factor for the subsequent linear interpolation in time. In order for this to
1047 ! work, the array holStru%castTime should be ordered in ascending order.
1048 CALL getlocandratio(times(icnt), holstru(stcnt)%castTime, jl1, jl2, wtratio)
1049
1050 ! Skip the subsequent calculations if Times(iCnt) is outside the castTime range
1051 ! by exiting this loop
1052 IF ((jl1 <= 0) .OR. (jl2 <= 0)) THEN
1053 WRITE(scratchmessage, '(a)') 'Requested output time: ' // trim(adjustl(tmptimestr)) // &
1054 ', skipping generating data for this time'
1055 CALL logmessage(info, scratchmessage)
1056
1057 EXIT
1058 END IF
1059
1060 ! Perform linear interpolation in time
1061 stormnumber = holstru(stcnt)%stormNumber(jl1)
1062
1063 CALL sphericalfracpoint(holstru(stcnt)%lat(jl1), holstru(stcnt)%lon(jl1), &
1064 holstru(stcnt)%lat(jl2), holstru(stcnt)%lon(jl2), &
1065 wtratio, lat, lon)
1066 !lat = holStru(stCnt)%lat(jl1) + &
1067 ! wtRatio * (holStru(stCnt)%lat(jl2) - holStru(stCnt)%lat(jl1))
1068 !lon = holStru(stCnt)%lon(jl1) + &
1069 ! wtRatio * (holStru(stCnt)%lon(jl2) - holStru(stCnt)%lon(jl1))
1070
1071 ! Radius of the last closed isobar
1072 rrp = holstru(stcnt)%rrp(jl1) + &
1073 wtratio * (holstru(stcnt)%rrp(jl2) - holstru(stcnt)%rrp(jl1))
1074
1075 ! Radius of maximum winds
1076 rmw = holstru(stcnt)%rmw(jl1) + &
1077 wtratio * (holstru(stcnt)%rmw(jl2) - holstru(stcnt)%rmw(jl1))
1078
1079 ! Get all the distances of the mesh nodes from (lat, lon)
1080 rad = sphericaldistance(sfea, slam, lat, lon)
1081 ! ... and the indices of the nodal points where rad <= rrp
1082 radidx = pack([(i, i = 1, np)], rad <= rrp)
1083 maxradidx = SIZE(radidx)
1084
1085 ! If the condition rad <= rrp is not satisfied anywhere then exit this loop
1086 IF (maxradidx == 0) THEN
1087 WRITE(tmpstr1, '(f20.3)') rrp
1088 tmpstr1 = '(rrp = ' // trim(adjustl(tmpstr1)) // ' m)'
1089 WRITE(scratchmessage, '(a)') 'No nodal points found inside the radius of the last closed isobar ' // &
1090 trim(adjustl(tmpstr1)) // ' for storm: ' // &
1091 trim(adjustl(holstru(stcnt)%thisStorm))
1092 CALL logmessage(info, scratchmessage)
1093
1094 EXIT
1095 END IF
1096
1097 speed = holstru(stcnt)%speed(jl1) + &
1098 wtratio * (holstru(stcnt)%speed(jl2) - holstru(stcnt)%speed(jl1))
1099
1100 cpress = holstru(stcnt)%cPress(jl1) + &
1101 wtratio * (holstru(stcnt)%cPress(jl2) - holstru(stcnt)%cPress(jl1))
1102
1103 trvx = holstru(stcnt)%trVx(jl1) + &
1104 wtratio * (holstru(stcnt)%trVx(jl2) - holstru(stcnt)%trVx(jl1))
1105 trvy = holstru(stcnt)%trVy(jl1) + &
1106 wtratio * (holstru(stcnt)%trVy(jl2) - holstru(stcnt)%trVy(jl1))
1107
1108 ! If this is a "CALM" period, set winds to zero velocity and pressure equal to the
1109 ! background pressure and return. PV: check if this is actually needed
1110 IF (cpress < 0.0_sz) THEN
1111 wpress(:, icnt) = backgroundatmpress * mb2pa
1112 wvelx(:, icnt) = 0.0_sz
1113 wvely(:, icnt) = 0.0_sz
1114
1115 WRITE(scratchmessage, '(a)') 'Calm period found, generating zero atmospheric fields for this time'
1116 CALL logmessage(info, scratchmessage)
1117
1118 EXIT
1119 END IF
1120
1121 ! Calculate and limit central pressure deficit; some track files (e.g., Charley 2004)
1122 ! may have a central pressure greater than the ambient pressure that this subroutine assumes
1123 cpressdef = backgroundatmpress * mb2pa - cpress
1124 IF (cpressdef < 100.0_sz) cpressdef = 100.0_sz
1125
1126 ! Subtract the translational speed of the storm from the observed max wind speed to avoid
1127 ! distortion in the Holland curve fit. The translational speed will be added back later.
1128 trspd = sqrt(trvx * trvx + trvy * trvy)
1129 speed = speed - trspd
1130
1131 ! Convert wind speed from 10 meter altitude (which is what the
1132 ! NHC forecast contains) to wind speed at the top of the atmospheric
1133 ! boundary layer (which is what the Holland curve fit requires).
1134 speed = speed / bladjustfac
1135
1136 ! Calculate Holland parameters and limit the result to its appropriate range.
1137 hlb = rhoair * basee * (speed**2) / cpressdef
1138 IF (hlb < 1.0_sz) hlb = 1.0_sz
1139 IF (hlb > 2.5_sz) hlb = 2.5_sz
1140
1141 ! If we are running storm 2 in the Lake Pontchartrain !PV Do we need this?
1142 ! Forecast System ensemble, the final wind speeds should be multiplied by 1.2.
1143 windmultiplier = 1.0_sz
1144 IF (stormnumber == 2) windmultiplier = 1.2_sz
1145
1146 DO npcnt = 1, maxradidx
1147 i = radidx(npcnt)
1148
1149 dx = sphericaldistance(lat, lon, lat, slam(i))
1150 dy = sphericaldistance(lat, lon, sfea(i), lon)
1151 theta = atan2(dy, dx)
1152
1153 ! Compute coriolis
1154 coriolis = 2.0_sz * omega * sin(sfea(i) * deg2rad)
1155
1156 ! Compute the pressure (Pa) at a distance rad(i); all distances are in meters
1157 sfpress = cpress + cpressdef * exp(-(rmw / rad(i))**hlb)
1158
1159 ! Compute wind speed (speed - trSPD) at gradient level (m/s) and at a distance rad(i);
1160 ! all distances are in meters. Using absolute value for coriolis for Southern Hempisphere
1161 grvel = sqrt(speed**2 * (rmw / rad(i))**hlb * exp(1.0_sz - (rmw / rad(i))**hlb) + &
1162 (rad(i) * abs(coriolis) / 2.0_sz)**2) - &
1163 rad(i) * abs(coriolis) / 2.0_sz
1164
1165 ! Determine translation speed that should be added to final !PV CHECK ON THIS
1166 ! storm wind speed. This is tapered to zero as the storm wind tapers
1167 ! to zero toward the eye of the storm and at long distances from the storm.
1168 trspdx = (abs(grvel) / speed ) * trvx
1169 trspdy = (abs(grvel) / speed ) * trvy
1170
1171 ! Apply mutliplier for Storm #2 in LPFS ensemble.
1172 grvel = grvel * windmultiplier
1173
1174 ! Find the wind velocity components.
1175 sfvelx = -grvel * sin(theta)
1176 sfvely = grvel * cos(theta)
1177 !print *, sfVelX, sfVelY
1178 ! Convert wind velocity from the gradient level (top of atmospheric boundary layer)
1179 ! which, is what the Holland curve fit produces, to 10-m wind velocity.
1180 sfvelx = sfvelx * bladjustfac
1181 sfvely = sfvely * bladjustfac
1182 !print *, sfVelX, sfVelY
1183 ! Convert from 1-minute averaged winds to 10-minute averaged winds.
1184 sfvelx = sfvelx * one2ten
1185 sfvely = sfvely * one2ten
1186 !print *, sfVelX, sfVelY
1187 ! Add back the storm translation speed.
1188 sfvelx = sfvelx + trspdx
1189 sfvely = sfvely + trspdy
1190
1191 !print *, sfVelX, sfVelY, wVelX(i, iCnt), wVelY(i, iCnt)
1192 !PV Need to interpolate between storms if this nodal point
1193 ! is affected by more than on storm
1194 wpress(i, icnt) = sfpress
1195 wvelx(i, icnt) = sfvelx
1196 wvely(i, icnt) = sfvely
1197
1198 !print *, sfVelX, sfVelY, wVelX(i, iCnt), wVelY(i, iCnt)
1199 !print *, '--------------------------------------'
1200 END DO ! npCnt = 1, maxRadIDX
1201
1202 END DO ! stCnt = 1, nBTrFiles
1203 END DO ! iCnt = 1, nOutDT
1204 WRITE(scratchmessage, '(a)') 'End of the main time loop'
1205 CALL allmessage(info, scratchmessage)
1206
1207 !---------- Deallocate the arrays
1208 IF (ALLOCATED(rad)) DEALLOCATE(rad)
1209 IF (ALLOCATED(radidx)) DEALLOCATE(radidx)
1210 DO icnt = 1, nbtrfiles
1211 CALL deallochollstruct(holstru(icnt))
1212 END DO
1213 DEALLOCATE(holstru)
1214 !----------
1215
1216 CALL unsetmessagesource()
1217
1218 END SUBROUTINE gethollandfields
