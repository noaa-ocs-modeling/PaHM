113 SUBROUTINE openlogfile()
114
115 IMPLICIT NONE
116
117 INTEGER :: errorIO ! zero if the file opened successfully
118
119 logfileopened = .false.
120
121 OPEN(unit=lun_log, file=trim(adjustl(logfilename)), action='WRITE', status='REPLACE', iostat=errorio)
122
123 IF (errorio == 0) THEN
124 logfileopened = .true.
125 ELSE
126 WRITE(scratchmessage, '(a, i0, a, i0)') &
127 'Could not open the log file = ' // trim(adjustl(logfilename)) // &
128 ' on logical unit LUN_LOG = ', lun_log, &
129 '. Error code was: errorIO = ', errorio
130 CALL screenmessage(error, scratchmessage)
131 END IF
132
133 END SUBROUTINE openlogfile
