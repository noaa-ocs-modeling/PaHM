268 USE pahm_global, ONLY : lun_btrk, lun_btrk1, nbtrfiles, besttrackfilename

275 CHARACTER(LEN=FNAMELEN) :: inpFile
276 CHARACTER(LEN=512) :: inpLine, line
277 CHARACTER(LEN=512) :: fmtStr
278
279 INTEGER :: lenLine
280 INTEGER :: nLines ! Number of lines counter
281 INTEGER :: iFile, iCnt ! loop counters
282 INTEGER :: iUnit, errIO, ios, status
283
284 !CHARACTER(LEN=4) :: castType
285
286 CHARACTER(LEN=10), ALLOCATABLE :: chkArrStr(:)

293 !---------- Initialize variables
294 iunit = lun_btrk
295 errio = 0
296
297 !fmtStr = '(a2, 2x, i2, 2x, a10, 2x, i2, 2x, a4, 2x, i3, 2x, i3, a1, 2x, i4, a1, 2x, i3, 2x, i4, 2x, a2, '
298 fmtstr = '(a2, 1x, i3, 2x, a10, 1x, i3, 2x, a4, 2x, i3, 2x, i3, a1, 1x, i5, a1, 2x, i3, 2x, i4, 2x, a2, '
299 fmtstr = trim(fmtstr) // ' 2x, i3, 2x, a3, 4(2x, i4), 2x, i4, 2x, i4, 2x, i3, 2x, i3, 2x, i3, '
300 fmtstr = trim(fmtstr) // ' 2x, a3, 2x, i3, 2x, a3, 1x, 2(i3, 2x), a11)'
301 !----------
302
303 CALL setmessagesource("ReadBestTrackFile")
304
305 ! Allocate the best track structure array. This structure holds all the
306 ! input values for the storm track as read in from the track input file
307 ! (a-deck, b-deck ATCF format) as well as the converted best track variables
308 ! (as appropriate).
309 ALLOCATE(besttrackdata(nbtrfiles))
310
311 ! This is the main loop. We loop through all the best track files
312 ! (user input)
313 DO ifile = 1, nbtrfiles
314 inpfile = besttrackfilename(ifile)
315
316 CALL openfileforread(iunit, trim(adjustl(inpfile)), errio)
317
318 IF (errio /= 0) THEN
319 WRITE(scratchmessage, '(a)') 'Error opening the best track file: ' // trim(adjustl(inpfile))
320 CALL allmessage(error, scratchmessage)
321
322 CALL unsetmessagesource()
323
324 CALL terminate()
325 ELSE
326 WRITE(scratchmessage, '(a)') 'Processing the best track file: ' // trim(adjustl(inpfile))
327 CALL logmessage(info, scratchmessage)
328 END IF
329
330 besttrackdata(ifile)%fileName = trim(adjustl(inpfile))
331 besttrackdata(ifile)%thisStorm = ""
332 besttrackdata(ifile)%loaded = .false.
333 besttrackdata(ifile)%numRec = -1
334
335 ! Count the number of non-empty or commented out lines in the file.
336 ! Comments are are considered those lines with the first non-blank character of "!" or "#"
337 nlines = 0
338 DO
339 READ(unit=iunit, fmt='(a)', err=10, END=5, IOSTAT=errIO) inpline
340
341 lenline = getlinerecord(inpline, line)
342 IF (lenline /= 0) nlines = nlines + 1
343 END DO
344 5 rewind(unit=iunit)
345
346 ! Array allocation in the structure bestTrackData
347 CALL allocbtrstruct(besttrackdata(ifile), nlines)
348
349 icnt = 0
350 kcnt = 0
351 DO WHILE (.true.)
352 READ(unit=iunit, fmt='(a)', err=10, END=20, IOSTAT=errIO) inpline
353
354 lenline = getlinerecord(inpline, line)
355
356 IF (lenline /= 0) THEN
357 icnt = icnt + 1
358 READ(line, fmt=fmtstr, err=11, iostat=ios) &
359 besttrackdata(ifile)%basin(icnt), besttrackdata(ifile)%cyNum(icnt), &
360 besttrackdata(ifile)%dtg(icnt), besttrackdata(ifile)%techNum(icnt), &
361 besttrackdata(ifile)%tech(icnt), besttrackdata(ifile)%tau(icnt), &
362 besttrackdata(ifile)%intLat(icnt), besttrackdata(ifile)%ns(icnt), &
363 besttrackdata(ifile)%intLon(icnt), besttrackdata(ifile)%ew(icnt), &
364 besttrackdata(ifile)%intVMax(icnt), besttrackdata(ifile)%intMslp(icnt), &
365 besttrackdata(ifile)%ty(icnt), besttrackdata(ifile)%rad(icnt), &
366 besttrackdata(ifile)%windCode(icnt), besttrackdata(ifile)%intRad1(icnt), &
367 besttrackdata(ifile)%intRad2(icnt), besttrackdata(ifile)%intRad3(icnt), &

369 besttrackdata(ifile)%intROuter(icnt), besttrackdata(ifile)%intRmw(icnt), &

372 besttrackdata(ifile)%initials(icnt), besttrackdata(ifile)%dir(icnt), &
373 besttrackdata(ifile)%intSpeed(icnt), besttrackdata(ifile)%stormName(icnt)
374
375 !---------- This is for the cycleNum, the last column we consider
376 IF (icnt == 1) THEN
377 kcnt = icnt
378 besttrackdata(ifile)%cycleNum(icnt) = kcnt
379 ELSE
380 kcnt = kcnt + 1
381 IF (besttrackdata(ifile)%dtg(icnt) == besttrackdata(ifile)%dtg(icnt-1)) THEN
382 besttrackdata(ifile)%cycleNum(icnt) = besttrackdata(ifile)%cycleNum(icnt-1)
383 kcnt = kcnt - 1
384 ELSE
385 besttrackdata(ifile)%cycleNum(icnt) = kcnt
386 END IF
387 END IF
388 !----------
389
390 !---------- Convert lat/lon values to S/N and W/E notations
391 IF (touppercase(besttrackdata(ifile)%ns(icnt)) == 'S') THEN
392 besttrackdata(ifile)%lat(icnt) = -0.1_sz * besttrackdata(ifile)%intLat(icnt)
393 ELSE
394 besttrackdata(ifile)%lat(icnt) = 0.1_sz * besttrackdata(ifile)%intLat(icnt)
395 END IF
396
397 IF (touppercase(besttrackdata(ifile)%ew(icnt)) == 'W') THEN
398 besttrackdata(ifile)%lon(icnt) = -0.1_sz * besttrackdata(ifile)%intLon(icnt)
399 ELSE
400 besttrackdata(ifile)%lon(icnt) = 0.1_sz * besttrackdata(ifile)%intLon(icnt)
401 END IF
402 !----------
403
404 !---------- Get the year, month, day, hour from the DGT string
405 READ(besttrackdata(ifile)%dtg(icnt)(1:4), fmt='(i4.4)', iostat=ios) besttrackdata(ifile)%year(icnt)
406 IF (ios /= 0) besttrackdata(ifile)%year(icnt) = -1
407 READ(besttrackdata(ifile)%dtg(icnt)(5:6), fmt='(i2.2)', iostat=ios) besttrackdata(ifile)%month(icnt)
408 IF (ios /= 0) besttrackdata(ifile)%month(icnt) = -1
409 READ(besttrackdata(ifile)%dtg(icnt)(7:8), fmt='(i2.2)', iostat=ios) besttrackdata(ifile)%day(icnt)
410 IF (ios /= 0) besttrackdata(ifile)%day(icnt) = -1
411 READ(besttrackdata(ifile)%dtg(icnt)(9:10), fmt='(i2.2)', iostat=ios) besttrackdata(ifile)%hour(icnt)
412 IF (ios /= 0) besttrackdata(ifile)%hour(icnt) = -1
413 !----------
414 END IF
415 END DO
416
417 10 IF (errio /= 0) THEN
418 WRITE(scratchmessage, '(a)') 'Error in file: ' // trim(adjustl(inpfile)) // &
419 ', while processing line: ' // trim(adjustl(inpline))
420 CALL allmessage(error, scratchmessage)
421
422 CLOSE(iunit)
423
424 CALL unsetmessagesource()
425
426 CALL terminate()
427 END IF
428
429 11 IF (ios /= 0) THEN
430 WRITE(scratchmessage, '(a)') 'Error in file: ' // trim(adjustl(inpfile)) // &
431 ', while processing line: ' // trim(adjustl(line))
432 CALL allmessage(error, scratchmessage)
433
434 CLOSE(iunit)
435
436 CALL unsetmessagesource()
437
438 CALL terminate()
439 END IF
440
441 20 CLOSE(iunit)
442
443 besttrackdata(ifile)%thisStorm = ''
444 besttrackdata(ifile)%loaded = .true.
445 besttrackdata(ifile)%numRec = nlines
446
447
448 !------------------------------------------------------------
449 ! Get the unique storm name and store it in the thisStorm string
450 ALLOCATE(chkarrstr(nlines))
451 ALLOCATE(idxarrstr(nlines))
452
453 nunique = charunique(besttrackdata(ifile)%stormName, chkarrstr, idxarrstr)
454
455 maxcnt = -1
456 DO kcnt = 1, nunique
457 kmax = count(chkarrstr(kcnt) == besttrackdata(ifile)%stormName)
458 IF (kmax > maxcnt) THEN
459 maxcnt = kmax
460 besttrackdata(ifile)%thisStorm = trim(adjustl(chkarrstr(kcnt)))
461 END IF
462 END DO
463
464 DEALLOCATE(chkarrstr)
465 DEALLOCATE(idxarrstr)
466 !------------------------------------------------------------
467
468 !------------------------------------------------------------
469 ! This is an extra step (paranoid) to ensure that the dates in the bestTrackData are
470 ! stored in ascending order
471 ALLOCATE(idx0(besttrackdata(ifile)%numRec))
472 ALLOCATE(idx1(besttrackdata(ifile)%numRec))
473
474 CALL indexx(besttrackdata(ifile)%dtg, idx1, status, .true.)
475
476 IF (status /= 0) THEN
477 CALL unsetmessagesource()
478
479 CALL terminate()
480 END IF
481
482 ! Create the index array to be used in the comparison below
483 idx0 = arth(1, 1, besttrackdata(ifile)%numRec)
484
485 IF (.NOT. arrayequal(idx0, idx1)) THEN
486 besttrackdata(ifile)%basin = besttrackdata(ifile)%basin(idx1)
487 besttrackdata(ifile)%cyNum = besttrackdata(ifile)%cyNum(idx1)
488 besttrackdata(ifile)%dtg = besttrackdata(ifile)%dtg(idx1)
489 besttrackdata(ifile)%techNum = besttrackdata(ifile)%techNum(idx1)
490 besttrackdata(ifile)%tech = besttrackdata(ifile)%tech(idx1)
491 besttrackdata(ifile)%tau = besttrackdata(ifile)%tau(idx1)
492 besttrackdata(ifile)%intLat = besttrackdata(ifile)%intLat(idx1)
493 besttrackdata(ifile)%ns = besttrackdata(ifile)%ns(idx1)
494 besttrackdata(ifile)%intLon = besttrackdata(ifile)%intLon(idx1)
495 besttrackdata(ifile)%ew = besttrackdata(ifile)%ew(idx1)
496 besttrackdata(ifile)%intVMax = besttrackdata(ifile)%intVMax(idx1)
497 besttrackdata(ifile)%intMslp = besttrackdata(ifile)%intMslp(idx1)
498 besttrackdata(ifile)%ty = besttrackdata(ifile)%ty(idx1)
499 besttrackdata(ifile)%rad = besttrackdata(ifile)%rad(idx1)
500 besttrackdata(ifile)%windCode = besttrackdata(ifile)%windCode(idx1)
501 besttrackdata(ifile)%intRad1 = besttrackdata(ifile)%intRad1(idx1)
502 besttrackdata(ifile)%intRad2 = besttrackdata(ifile)%intRad2(idx1)
503 besttrackdata(ifile)%intRad3 = besttrackdata(ifile)%intRad3(idx1)
