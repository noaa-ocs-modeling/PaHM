137

141

143
144 INTEGER, INTENT(IN), OPTIONAL :: nTimeSTP
145
146 INTEGER :: iCnt
147
148 CALL setmessagesource("PaHM_Run")
149
150 IF (PRESENT(ntimestp)) THEN
151 cnttimeend = cnttimebegin + ntimestp - 1
152 ENDIF
153
154 SELECT CASE (modeltype)
155 CASE (1)
156 DO icnt = cnttimebegin, cnttimeend
157 CALL gethollandfields(icnt)
158 IF (outfilenamespecified) THEN
159 ! Create the output NetCDF file and fill it with the static data only
160 ! Initialize all variables. This subroutine is called just once
161 CALL initadcircnetcdfoutfile(outfilename)
162
163 CALL writenetcdfrecord(outfilename, icnt)
164 END IF
165 END DO
166
167 CASE DEFAULT
168 WRITE(scratchmessage, '(a, i0)') &
169 'This model type is not supported: modelType = ', modeltype
170 CALL logmessage(error, scratchmessage)
171 END SELECT
172
173 IF (PRESENT(ntimestp)) THEN
174 cnttimebegin = cnttimeend + 1
175 ENDIF
176
177 CALL unsetmessagesource()
178
