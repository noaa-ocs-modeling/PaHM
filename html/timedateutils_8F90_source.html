315 LOGICAL FUNCTION leapyear(iYear) RESULT(myVal)
316
317 IMPLICIT NONE
318
319 INTEGER, INTENT(IN) :: iyear
320
321 !----- START CALCULATIONS -----
322
323 IF (iyear < 1582) Then
324 myval = .false.
325
326 RETURN
327 END IF
328
329 ! ADCIRC uses the construct leap = (iYear / 4) * 4 == iYear
330 ! to determine if a year is a leap year. This produces wrong
331 ! results, example while 1700, 1900, 2100 are not leap years,
332 ! the above construct determines that these years are leap years.
333 ! Needs to be fixed.
334
335 IF ((mod(iyear, 100) /= 0) .AND. (mod(iyear, 4) == 0)) THEN
336 myval = .true.
337 ELSE IF (mod(iyear, 400) == 0) THEN
338 myval = .true.
339 ELSE
340 myval = .false.
341 END IF
342
343 RETURN
344 END FUNCTION leapyear

536 REAL(sz) function gregtojuldayisec(iyear, imonth, iday, ihour, imin, isec, mjd) result(myval)
537
538 IMPLICIT NONE
539
540 ! Global variables
541 INTEGER, INTENT(IN) :: iyear, imonth, iday, ihour, imin, isec
542 INTEGER, OPTIONAL, INTENT(IN) :: mjd
543
544 ! Local variables
545 INTEGER :: leap, monlen(12, 2)
546 LOGICAL :: modjul
547 REAL(hp) :: temp1, temp2
548
549 !----- START CALCULATIONS -----
550
551 modjul = .false.
552 IF (PRESENT(mjd)) THEN
553 modjul = (mjd > 0)
554 ELSE
555 modjul = (usemodjulday > 0)
556 END IF
557
558 ! Initialize lenghts of months:
559 monlen = reshape((/ 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31, &

561 (/ 12, 2 /))
562
563 ! This function intentionally works on Gregorian dates only. For modeling
564 ! purposes the min date supported 1582/10/05 is sufficient. Most likely,
565 ! it is not necessary to go beyond that date.
566
567 ! Is this a LEAP year?
568 leap = 1
569 IF (leapyear(iyear)) leap = 2
570
571 IF (joindate(iyear, imonth, iday) < firstgregdate) THEN
572 myval = rmissv
573
574 RETURN
575 ELSE IF ((imonth < 1) .OR. (imonth > 12) .OR. &
576 (iday < 1) .OR. (iday > monlen(imonth, leap)) .OR. &

579 (isec < 0) .OR. (isec > 60)) THEN
580 myval = rmissv
581
582 RETURN
583 ELSE
584 temp1 = int((imonth - 14.0_hp) / 12.0_hp)
585 temp2 = iday - 32075.0_hp &
586 + int(1461.0_hp * (iyear + 4800.0_hp + temp1) / 4.0_hp) &
587 + int(367.0_hp * (imonth - 2.0_hp - temp1 * 12.0_hp) / 12.0_hp) &
588 - int(3.0_hp * int((iyear + 4900.0_hp + temp1) / 100.0_hp) / 4.0_hp)
589 temp1 = real(ihour, hp) * 3600.0_hp &
590 + real(imin, hp) * 60.0_hp &
591 + real(isec, hp) - 43200.0_hp
592
593 IF (modjul) THEN
594 print *, 'we are using mod jul with MDJOFFSET = ', mdjoffset
595 myval = temp2 + (temp1 / 86400.0_hp) - mdjoffset
596 ELSE
597 myval = temp2 + (temp1 / 86400.0_hp)
598 END IF
599 END IF
600
601 RETURN
602 END FUNCTION gregtojuldayisec

899 SUBROUTINE juldaytogreg(julDay, iYear, iMonth, iDay, iHour, iMin, iSec, mJD)
900
901 IMPLICIT NONE
902
903 ! Global Variables
904 REAL(SZ), INTENT(IN) :: julDay
905 INTEGER, OPTIONAL, INTENT(IN) :: mJD
906 INTEGER, INTENT(OUT) :: iYear, iMonth, iDay, iHour, iMin, iSec
907
908 ! Local Variables
909 REAL(HP) :: temp1 , temp2 , temp3 , temp4 , temp5
910 REAL(HP) :: thisJulDay, myJulDay, delta
911 INTEGER :: nTry
912 LOGICAL :: modJul
913
914 !----- START CALCULATIONS -----
915
916 modjul = .false.
917 IF (PRESENT(mjd)) THEN
918 modjul = (mjd > 0)
919 ELSE
920 modjul = (usemodjulday > 0)
921 END IF
922
923 IF (modjul) THEN
924 thisjulday = julday + mdjoffset
925 ELSE
926 thisjulday = julday
927 END IF
928
929 ! Check for valid Julian day (Gregorian calendar only)
930 IF (thisjulday < offfirstgregday) THEN
931 iyear = imissv
932 imonth = imissv
933 iday = imissv
934 ihour = imissv
935 imin = imissv
936 isec = imissv
937
938 RETURN
939 END IF
940
941 delta = 0.0_hp
942 ntry = 1
943 DO WHILE (ntry <= 2)
944 myjulday= thisjulday + delta
945 temp4 = myjulday
946 temp5 = dmod(myjulday, 1.0_hp)
947
948 IF (temp5 < 0.5) THEN
949 temp3 = 0.5_hp + temp5
950 temp4 = aint(temp4)
951 ELSE
952 temp3 = temp5 - 0.5_hp
953 temp4 = aint(temp4) + 1.0_hp
954 END IF
955
956 temp1 = temp4 + 68569.0
957 temp2 = aint(4.0_hp * temp1 / 146097.0_hp)
958 temp1 = temp1 - aint((146097.0_hp * temp2 + 3.0_hp) / 4.0_hp)
959 iyear = int(4000.0_hp * (temp1 + 1.0_hp) / 1461001.0_hp)
960 temp1 = temp1 - aint((1461.0_hp * iyear) / 4.0_hp) + 31.0_hp
961 imonth = int(80.0_hp * temp1 / 2447.0_hp)
962 iday = int(temp1 - aint(2447.0_hp * imonth / 80.0_hp))
963 temp1 = aint(imonth / 11.0_hp)
964 imonth = int(imonth + 2.0 - 12.0_hp * temp1)
965 iyear = int(100.0_hp * (temp2 - 49.0_hp) + iyear + temp1)
966 ihour = int(temp3 * 24.0_hp)
967 imin = int(temp3 * 1440.0_hp - 60.0_hp * ihour)
968 isec = nint(temp3 * 86400.0_hp - 3600.0_hp * ihour - 60.0_hp * imin)
969
970 IF (isec >= 60) THEN
971 IF (ntry < 2) THEN
972 delta = 0.49999_hp / 86400.0_hp
973 ntry = ntry + 1
974 ELSE
975 iyear = imissv
