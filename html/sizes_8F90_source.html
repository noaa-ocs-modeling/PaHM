101 INTEGER FUNCTION comparedoublereals(rVal1, rVal2, eps) RESULT(myValOut)
102
103 IMPLICIT NONE
104
105 ! Global variables
106 REAL(hp), INTENT(IN) :: rval1, rval2
107 REAL(hp), OPTIONAL, INTENT(IN) :: eps
108
109 ! Local variables
110 REAL(hp) :: epssys, epsusr, value
111
112
113 epssys = 2.0_hp * epsilon(rval1)
114
115 IF (PRESENT(eps)) THEN
116 epsusr = abs(eps)
117 ELSE
118 epsusr = epssys
119 ENDIF
120
121 IF ((abs(rval1) < 1.0_hp) .OR. (abs(rval2) < 1.0_hp)) THEN
122 value = rval1 - rval2
123 ELSE
124 value = (rval1 - rval2) / max(rval1, rval2)
125 IF (abs(value) < 1.0_hp) value = rval1 - rval2
126 END IF
127
128 IF (abs(value) < epsusr) THEN
129 myvalout = 0
130 ELSE IF (rval1 < rval2) THEN
131 myvalout = -1
132 ELSE
133 myvalout = 1
134 END IF
135
136 RETURN
137
