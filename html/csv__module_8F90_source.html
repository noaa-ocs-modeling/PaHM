232 subroutine read_csv_file(me,filename,header_row,skip_rows,status_ok)
233
234 implicit none
235
236 class(csv_file),intent(inout) :: me
237 character(len=*),intent(in) :: filename
238 logical,intent(out) :: status_ok
239 integer,intent(in),optional :: header_row
240 integer,dimension(:),intent(in),optional :: skip_rows
241
242 type(csv_string),dimension(:),allocatable :: row_data
243 type(csv_string) :: empty_data
244 integer,dimension(:),allocatable :: rows_to_skip
245 character(len=:),allocatable :: line
246 integer :: i
247 integer :: j
248 integer :: irow
249 integer :: n_rows_in_file
250 integer :: n_rows
251 integer :: n_cols
252 integer :: istat
253 integer :: line_n_cols
254 integer :: iunit
255 logical :: arrays_allocated
257 integer :: iheader
259 character(len=1) :: tmp
260
261 empty_data%str = ' '
262 iunit = lun_btrk
263
264 CALL setmessagesource("read_csv_file")
265
266 call me%destroy()
267 arrays_allocated = .false.
268
269 CALL openfileforread(iunit, trim(adjustl(filename)), istat)
270
271 IF (istat /= 0) THEN
272 WRITE(scratchmessage, '(a)') 'Error opening the file: ' // trim(adjustl(filename))
273 CALL allmessage(error, scratchmessage)
274
275 CALL unsetmessagesource()
276
277 CALL terminate()
278 ELSE
279 WRITE(scratchmessage, '(a)') 'Processing the file: ' // trim(adjustl(filename))
280 CALL logmessage(info, scratchmessage)
281 END IF
282
283! if (istat==0) then
284
285 !get number of lines in the file
286 n_rows_in_file = number_of_lines_in_file(iunit)
287
288 !get number of lines in the data array
289 if (present(skip_rows)) then
290 !get size of unique elements in skip_rows,
291 !and subtract from n_rows_in_file
292 rows_to_skip = unique(skip_rows,chunk_size=me%chunk_size)
293 n_rows = n_rows_in_file - size(rows_to_skip)
294 else
295 n_rows = n_rows_in_file
296 end if
297 if (present(header_row)) then
298 iheader = max(0,header_row)
299 n_rows = n_rows - 1
300 else
301 iheader = 0
302 end if
303
304 me%n_rows = n_rows
305
306 ! we don't know the number of columns
307 ! until we parse the first row (or the header)
308 ! Panagiotis Velissariou: some csv files do not have the same number
309 ! of columns, so we need to determine the nax number of columns
310 ! for the allocation of the arrays
311 !--- PV
312 n_cols = 0
313 do i=1,n_rows_in_file
314 call me%read_line_from_file(iunit,line,status_ok)
315 call me%tokenize(line,row_data)
316 n_cols = max(n_cols,size(row_data))
317 end do
318 rewind(iunit)
319
320 me%n_cols = n_cols
321 allocate(me%csv_data(n_rows,n_cols))
322 if (iheader/=0) allocate(me%header(n_cols))
323 arrays_allocated = .true.
324 !--- PV
325
326 !read each line in the file, parse it, and populate data
327 irow = 0
328 do i=1,n_rows_in_file
329
330 ! skip row if necessary
331 if (allocated(rows_to_skip)) then
332 if (any(i==rows_to_skip)) then
333 read(iunit,fmt='(A1)',iostat=istat) tmp
334 if (istat/=0) then
335 scratchmessage = 'Error skipping row in file: '//trim(filename)
336 CALL allmessage(error, scratchmessage)
337
338 close(unit=iunit,iostat=istat)
339 status_ok = .false.
340
341 CALL unsetmessagesource()
342 return
343 end if
344 cycle
345 end if
346 end if
347
348 call me%read_line_from_file(iunit,line,status_ok)
349 if (.not. status_ok) then
350 CALL unsetmessagesource()
351 return ! file read error
352 end if
353 call me%tokenize(line,row_data)
354 line_n_cols = size(row_data)
355
356 if (i==iheader) then
357 do j=1,me%n_cols
358 me%header(j)%str = row_data(j)%str
359 end do
360 else
361 irow = irow + 1
362 do j=1,n_cols
363 if(j <= line_n_cols) then
364 me%csv_data(irow,j) = row_data(j) !%str
365 else
366 me%csv_data(irow,j) = empty_data !%str
367 end if
368 end do
369 end if
370
371 end do
372
373 ! close the file
374 close(unit=iunit,iostat=istat)
375
376 status_ok = .true.
377
378! else
379! scratchMessage = 'Error opening file: '//trim(filename)
380! CALL AllMessage(ERROR, scratchMessage)
381! status_ok = .false.
382! end if
383
384 CALL unsetmessagesource()
385
386 end subroutine read_csv_file

396
397 implicit none
398
399 class(csv_file),intent(inout) :: me
400 character(len=*),intent(in) :: filename
401 integer,intent(in) :: n_cols
402 logical,intent(out) :: status_ok
403 logical,intent(in),optional :: append
404
405 integer :: istat
406 logical :: append_flag
407 logical :: file_exists
408
409 CALL setmessagesource("open_csv_file")
410
411 call me%destroy()
412
413 me%n_cols = n_cols
414
415 ! optional append argument:
416 append_flag = .false.
417 file_exists = .false.
418 if (present(append)) then
419 append_flag = append
420 if (append) inquire(file=filename, exist=file_exists)
421 end if
422
423 if (append_flag .and. file_exists) then
424 open(unit=me%iunit,file=filename,status='OLD',position='APPEND',iostat=istat)
425 else
426 open(unit=me%iunit,file=filename,status='REPLACE',iostat=istat)
427 end if
428
429 if (istat==0) then
430 status_ok = .true.
431 else
432 scratchmessage = 'Error opening file: '//trim(filename)
433 CALL allmessage(error, scratchmessage)
434 status_ok = .false.
435 end if
436
437 CALL unsetmessagesource()
438
439 end subroutine open_csv_file

467 subroutine add_cell(me,val,int_fmt,real_fmt,trim_str)
468
469 implicit none
470
471 class(csv_file),intent(inout) :: me
472 class(*),intent(in) :: val
473 character(len=*),intent(in),optional :: int_fmt
475 character(len=*),intent(in),optional :: real_fmt
477 logical,intent(in),optional :: trim_str
478
479 integer :: istat
480 character(len=:),allocatable :: ifmt
481 character(len=:),allocatable :: rfmt
482 logical :: trimstr
483 character(len=max_real_str_len) :: real_val
484 character(len=max_integer_str_len) :: int_val
485
486 CALL setmessagesource("add_cell")
487
488 ! make sure the row isn't already finished
489 if (me%icol<me%n_cols) then
490
491 me%icol = me%icol + 1
492
493 if (me%enclose_all_in_quotes) then
494 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) me%quote
495 end if
496
497 select type (val)
498 type is (integer(ip))
499 if (present(int_fmt)) then
500 ifmt = trim(adjustl(int_fmt))
501 else
502 ifmt = default_int_fmt
503 end if
504 write(int_val,fmt=ifmt,iostat=istat) val
505 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) trim(adjustl(int_val))
506 type is (real(wp))
507 if (present(real_fmt)) then
508 rfmt = trim(adjustl(real_fmt))
509 else
510 rfmt = default_real_fmt
511 end if
512 write(real_val,fmt=rfmt,iostat=istat) val
513 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) trim(adjustl(real_val))
514 type is (logical)
515 if (val) then
516 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) me%logical_true_string
517 else
518 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) me%logical_false_string
519 end if
520 type is (character(len=*))
521 if (me%enclose_strings_in_quotes .and. .not. me%enclose_all_in_quotes) &
522 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) me%quote
523 if (present(trim_str)) then
524 trimstr = trim_str
525 else
526 trimstr = .false.
527 end if
528 if (trimstr) then
529 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) trim(val)
530 else
531 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) val
532 end if
533 if (me%enclose_strings_in_quotes .and. .not. me%enclose_all_in_quotes) &
534 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) me%quote
535 type is (csv_string)
536 if (me%enclose_strings_in_quotes .and. .not. me%enclose_all_in_quotes) &
537 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) me%quote
538 if (present(trim_str)) then
539 trimstr = trim_str
540 else
541 trimstr = .false.
542 end if
543 if (trimstr) then
544 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) trim(val%str)
545 else
546 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) val%str
547 end if
548 if (me%enclose_strings_in_quotes .and. .not. me%enclose_all_in_quotes) &
549 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) me%quote
550 class default
551 scratchmessage = 'Error: cannot write unknown variable type to CSV file.'
552 CALL allmessage(error, scratchmessage)
553 end select
554
555 if (me%enclose_all_in_quotes) then
556 write(me%iunit,fmt='(A)',advance='NO',iostat=istat) me%quote
557 end if
558 if (me%icol<me%n_cols) write(me%iunit,fmt='(A)',advance='NO',iostat=istat) me%delimiter
559
560 else
561 scratchmessage = 'Error: cannot write more cells to the current row.'
562 CALL allmessage(error, scratchmessage)
563 end if
564
565 CALL unsetmessagesource()
566
567 end subroutine add_cell

641 subroutine next_row(me)
642
643 implicit none
644
645 class(csv_file),intent(inout) :: me
646
647 integer :: i
648 integer :: n
649
650 if (me%icol>0) then
651 n = me%n_cols - me%icol
652 do i=1,n
653 if (i==n) then !no trailing delimiter

662 end if
663 end if
664 end do
665 write(me%iunit,'(A)') '' ! new line
666 end if
